    """权重计算 - 基于身份证明"""
    
    def __init__(self):
        self.dna追溯码 = "#龍芯⚡️2026-01-30-权重计算器-v1.0"
    
    def 计算权重(self, 身份: Dict[str, any]) -> float:
        """
        计算用户权重（基于身份证明）
        
        公式: 权重 = 基础权重 + 身份加成 + 贡献加成 + 时间加成
        """
        # 1. 基础权重（人人平等）
        基础权重 = 1.0
        
        # 2. 身份加成（GPG验证 + 网络身份证）
        身份加成 = 0.0
        
        if 身份.get("GPG验证通过"):
            身份加成 += 2.0
        
        # 根据网络身份证前缀加成
        网络身份证 = 身份.get("网络身份证", "")
        if 网络身份证.startswith("T38"):  # 军人
            身份加成 += 5.0
        elif 网络身份证.startswith("T39"):  # 烈属
            身份加成 += 4.0
        elif 网络身份证.startswith("T40"):  # 贡献者
            身份加成 += 3.0
        
        # 3. 贡献加成（帮人行为）
        贡献加成 = 身份.get("帮人次数", 0) * 0.5
        
        # 4. 时间加成（注册时长）
        注册时间 = 身份.get("注册时间", datetime.now())
        注册天数 = (datetime.now() - 注册时间).days
        时间加成 = min(注册天数 / 365, 2.0)  # 最多2分
        
        # 总权重
        总权重 = 基础权重 + 身份加成 + 贡献加成 + 时间加成
        
        # 封顶10分
        return min(总权重, 10.0)
    
    def 计算权限等级(self, 权重: float) -> str:
        """根据权重计算权限等级"""
        if 权重 >= 9.0:
            return "先驱者"
        elif 权重 >= 7.0:
            return "守护者"
        elif 权重 >= 5.0:
            return "传承者"
        elif 权重 >= 3.0:
            return "贡献者"
        elif 权重 >= 2.0:
            return "活跃成员"
        else:
            return "新手"
    
    def 获取权限列表(self, 权重: float) -> list:
        """根据权重获取权限列表"""
        所有权限 = {
            "基础权限": ["使用权", "发言权", "投票权", "建议权", "传播权", "编辑权", "隐私权"],
            "提案权": 权重 >= 2.0,
            "代码编辑权": 权重 >= 3.0,
            "传承者权限": 权重 >= 5.0,
            "守护者权限": 权重 >= 7.0,
            "先驱者权限": 权重 >= 9.0
        }
        
        权限列表 = 所有权限["基础权限"].copy()
        
        for 权限, 条件 in 所有权限.items():
            if 权限 != "基础权限" and 条件:
                权限列表.append(权限)
        
        return 权限列表

if __name__ == "__main__":
    计算器 = 权重计算器()
    print("✅ 权重计算器创建成功")
    print(f"   DNA追溯码: {计算器.dna追溯码}")
