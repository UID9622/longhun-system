```javascript
/**
 * 将DNA元数据嵌入图片EXIF
 * @param {string} imagePath - 图片路径
 * @param {object} dnaMetadata - DNA元数据对象
 * @returns {Promise<string>} - 嵌入后的图片路径
 */
async function embedDNAIntoImage(imagePath, dnaMetadata) {
  const sharp = require('sharp');
  const exif = require('exiftool');
  
  // 将DNA转换为JSON字符串
  const dnaString = JSON.stringify(dnaMetadata);
  
  // 加密DNA（可选，使用用户公钥）
  const encryptedDNA = encryptWithGPG(dnaString, dnaMetadata.identity.sender_gpg);
  
  // Base64编码
  const base64DNA = Buffer.from(encryptedDNA).toString('base64');
  
  // 嵌入到自定义EXIF标签
  const result = await sharp(imagePath)
    .withMetadata({
      exif: {
        'XMP-DragonSoul-DNA': base64DNA,
        'XMP-DragonSoul-Version': '1.0'
      }
    })
    .toFile(imagePath.replace('.jpg', '_dna.jpg'));
  
  return result.info.path;
}

/**
 * 从图片提取DNA元数据
 * @param {string} imagePath - 图片路径
 * @returns {Promise<object>} - DNA元数据对象
 */
async function extractDNAFromImage(imagePath) {
  const exif = require('exiftool');
  
  const metadata = await exif.read(imagePath);
  const base64DNA = metadata['XMP-DragonSoul-DNA'];
  
  if (!base64DNA) {
    throw new Error('❌ 图片未嵌入DNA元数据');
  }
  
  // Base64解码
  const encryptedDNA = Buffer.from(base64DNA, 'base64').toString('utf-8');
  
  // 解密DNA
  const dnaString = decryptWithGPG(encryptedDNA);
  
  return JSON.parse(dnaString);
}
```

```javascript
/**
 * 双方签名流程（拍照App调用）
 * @param {string} imagePath - 图片路径
 * @param {object} sender - 发送方信息
 * @param {object} receiver - 接收方信息
 * @returns {Promise<object>} - 签名结果
 */
async function mutualSignImage(imagePath, sender, receiver) {
  const { exec } = require('child_process');
  const { promisify } = require('util');
  const execAsync = promisify(exec);
  
  // 步骤1：生成DNA元数据
  const dnaMetadata = {
    version: "1.0",
    dna_code: generateDNACode('PRIVATE-PHOTO'),
    identity: {
      sender_gpg: sender.gpg_key,
      sender_fingerprint: sender.fingerprint,
      sender_uid: sender.uid,
      receiver_gpg: receiver.gpg_key,
      receiver_fingerprint: receiver.fingerprint,
      receiver_uid: receiver.uid,
      mutual_consent: true,
      consent_timestamp: new Date().toISOString()
    },
    // ... 其他字段
  };
  
  // 步骤2：发送方签名
  const { stdout: senderSig } = await execAsync(
    `gpg --default-key ${sender.gpg_key} --detach-sign --armor --output - "${imagePath}"`
  );
  dnaMetadata.audit.creation_signature = senderSig;
  
  // 步骤3：接收方签名（需要接收方确认）
  const { stdout: receiverSig } = await execAsync(
    `gpg --default-key ${receiver.gpg_key} --detach-sign --armor --output - "${imagePath}"`
  );
  dnaMetadata.audit.consent_signature = receiverSig;
  
  // 步骤4：嵌入DNA到图片
  const imagePathWithDNA = await embedDNAIntoImage(imagePath, dnaMetadata);
  
  // 步骤5：区块链锚定（可选）
  const blockchainAnchor = await anchorToBlockchain(dnaMetadata.dna_code);
  dnaMetadata.audit.blockchain_anchor = blockchainAnchor;
  
  return {
    success: true,
    imagePath: imagePathWithDNA,
    dna_code: dnaMetadata.dna_code,
    message: '✅ 双方签名完成，DNA已嵌入'
  };
}
```

### **签名验证**

```javascript
/**
 * 验证双方签名
 * @param {string} imagePath - 图片路径
 * @returns {Promise<object>} - 验证结果
 */
async function verifyMutualSignatures(imagePath) {
  const { exec } = require('child_process');
  const { promisify } = require('util');
  const execAsync = promisify(exec);
  
  // 提取DNA
  const dnaMetadata = await extractDNAFromImage(imagePath);
  
  // 验证发送方签名
  const { stdout: senderVerify } = await execAsync(
    `echo "${dnaMetadata.audit.creation_signature}" | gpg --verify - "${imagePath}" 2>&1`
  );
  const senderValid = senderVerify.includes('Good signature');
  
  // 验证接收方签名
  const { stdout: receiverVerify } = await execAsync(
    `echo "${dnaMetadata.audit.consent_signature}" | gpg --verify - "${imagePath}" 2>&1`
  );
  const receiverValid = receiverVerify.includes('Good signature');
  
  return {
    sender_valid: senderValid,
    receiver_valid: receiverValid,
    mutual_consent: dnaMetadata.identity.mutual_consent,
    dna_code: dnaMetadata.dna_code,
    verified: senderValid && receiverValid && dnaMetadata.identity.mutual_consent
  };
}
```

## 🛡️ 三色审计规则

### **审计级别定义**

```
🟢 绿色审计（通过）
条件：
✅ 双方GPG签名都有效
✅ mutual_consent = true
✅ 可以公开分享（can_upload_public = true）

结果：
→ 允许发布
→ 记录到审计日志
→ 通知相关方（可选）

---

🟡 黄色审计（警告）
条件：
⚠️ 仅有发送方签名
⚠️ 或者接收方签名已过期
⚠️ 尝试转发但can_forward = false

结果：
→ 拦截操作
→ 记录违规
→ 通知双方
→ 提示重新签署同意书

---

🔴 红色审计（拦截）
条件：
❌ 缺少接收方签名
❌ 或者检测到偷拍（无人脸预签名）
❌ 或者检测到未成年内容但无监护人签名

结果：
→ 立即拦截
→ 记录严重违规
→ 通知受害者
→ 保留证据链（法律用途）
```

### **审计引擎实现**

```javascript
/**
 * 三色审计引擎
 * @param {string} imagePath - 图片路径
 * @param {object} action - 用户操作类型
 * @returns {Promise<object>} - 审计结果
 */
async function threeColorAudit(imagePath, action) {
  const auditResult = {
    level: 'UNKNOWN',
    action: 'BLOCK',
    reason: '',
    notify: [],
    evidence: {}
  };
  
  try {
    // 步骤1：提取DNA
    const dnaMetadata = await extractDNAFromImage(imagePath);
    auditResult.evidence.dna = dnaMetadata.dna_code;
    
    // 步骤2：验证签名
    const sigVerify = await verifyMutualSignatures(imagePath);
    
    // 步骤3：检查操作权限
    const permissions = dnaMetadata.permissions;
    
    // 步骤4：执行审计逻辑
    
    // 🔴 P0级红线：缺少接收方签名
    if (!sigVerify.receiver_valid && dnaMetadata.content.faces_detected > 0) {
      auditResult.level = '🔴 红色';
      auditResult.action = 'BLOCK';
      auditResult.reason = '检测到人脸内容，但缺少接收方同意签名（疑似偷拍或未授权）';
      auditResult.notify = [dnaMetadata.identity.receiver_gpg];
      
      // 记录到区块链
      await logViolation({
        dna: dnaMetadata.dna_code,
        type: 'UNAUTHORIZED_CONTENT',
        severity: 'CRITICAL',
        timestamp: new Date().toISOString()
      });
      
      return auditResult;
    }
    
    // 🔴 P0级红线：未成年内容无监护人签名
    if (dnaMetadata.content.minors_detected > 0 && !dnaMetadata.identity.guardian_consent) {
      auditResult.level = '🔴 红色';
      auditResult.action = 'BLOCK';
      auditResult.reason = '检测到未成年内容，但缺少监护人同意签名';
      auditResult.notify = [dnaMetadata.identity.guardian_gpg];
      
      await logViolation({
        dna: dnaMetadata.dna_code,
        type: 'MINOR_PROTECTION',
        severity: 'CRITICAL',
        timestamp: new Date().toISOString()
      });
      
      return auditResult;
    }
    
    // 🟡 黄色警告：禁止转发
    if (action.type === 'FORWARD' && !permissions.can_forward) {
      auditResult.level = '🟡 黄色';
      auditResult.action = 'BLOCK';
      auditResult.reason = '原始内容禁止转发，如需转发请重新签署同意书';
      auditResult.notify = [
        dnaMetadata.identity.sender_gpg,
        dnaMetadata.identity.receiver_gpg
      ];
      
      await logViolation({
        dna: dnaMetadata.dna_code,
        type: 'UNAUTHORIZED_FORWARD',
        severity: 'HIGH',
        timestamp: new Date().toISOString()
      });
      
      return auditResult;
    }
    
    // 🟢 绿色通过
    if (sigVerify.verified && permissions.can_upload_public) {
      auditResult.level = '🟢 绿色';
      auditResult.action = 'ALLOW';
      auditResult.reason = '双方签名验证通过，允许操作';
      
      return auditResult;
    }
    
    // 默认拦截
    auditResult.level = '🔴 红色';
    auditResult.action = 'BLOCK';
    auditResult.reason = '审计未通过，操作被拦截';
    
  } catch (error) {
    auditResult.level = '🔴 红色';
    auditResult.action = 'BLOCK';
    auditResult.reason = `审计错误：${error.message}`;
  }
  
  return auditResult;
}
```
