## ⚠️ 重要声明

**本系统设计理念：**
- ✅ 保护人民群众财产安全
- ✅ 预防电信网络诈骗
- ✅ 教育引导无知群众
- ✅ 协助公安机关反诈工作

**技术限制：**
- ❌ 不直接接入公安内网（涉密网络）
- ❌ 不侵犯用户隐私
- ❌ 不监控正常通信
- ✅ 仅本地关键词预警
- ✅ 用户自愿安装
- ✅ 预警信息本地处理

## 🎯 系统架构

### 合法合规的技术路径

```
用户本地设备
    │
    ├─ 本地AI监听（用户授权）
    │   └─ 精准关键词匹配（本地库）
    │
    ├─ 触发预警
    │   └─ 本地弹窗教育
    │   └─ 播放反诈音频
    │   └─ 显示案例警示
    │
    └─ 自愿上报（用户同意）
        └─ 脱敏数据 → 公安反诈平台API（公开接口）
```

## 🛠️ 本地反诈预警系统实现

### 核心功能：本地关键词预警

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
龍魂反诈预警系统 - 本地守护版
在用户设备本地运行，不上传隐私数据
"""

import re
import json
import hashlib
from datetime import datetime
from typing import Dict, List, Optional

class 龍魂反诈预警系统:
    """
    本地反诈骗预警系统
    保护无知群众不上当受骗
    """
    
    # 诈骗关键词库（本地维护）
    诈骗关键词库 = {
        "投资诈骗": [
            "高回报", "稳赚不赔", "内幕消息", "保本高收益",
            "杀猪盘", "荐股", "虚拟货币", "区块链投资"
        ],
        "冒充公检法": [
            "涉嫌洗钱", "通缉令", "安全账户", "资金审查",
            "保密案件", "不能告诉家人", "远程笔录"
        ],
        "网贷诈骗": [
            "无抵押贷款", "低息贷款", "保证金", "解冻费",
            "刷流水", "征信修复", "包装费"
        ],
        "刷单诈骗": [
            "兼职刷单", "日赚千元", "动动手指", "垫付",
            "连单", "任务未完成", "无法提现"
        ],
        "冒充客服": [
            "商品质量问题", "退款", "多倍赔偿", "备用金",
            "关闭会员", "否则扣费", "下载APP"
        ],
        "情感诈骗": [
            "高富帅", "白富美", "异国恋", "投资平台",
            "带你赚钱", "共度余生", "网恋投资"
        ]
    }
    
    # 预警等级
    预警等级 = {
        "低危": {"颜色": "🟡", "动作": "提示"},
        "中危": {"颜色": "🟠", "动作": "警告"},
        "高危": {"颜色": "🔴", "动作": "阻断+教育"}
    }
    
    def __init__(self):
        self.DNA追溯码 = "#龍芯⚡️2026-02-06-龍魂反诈系统-v1.0"
        self.预警记录 = []
        self.用户授权 = False  # 必须用户同意
    
    def 用户授权协议(self) -> bool:
        """
        显示用户授权协议
        必须用户明确同意才能启用监控
        """
        协议 = """
        ═══════════════════════════════════════════════════════════
                        龍魂反诈预警系统 - 用户授权协议
        ═══════════════════════════════════════════════════════════
        
        本系统旨在保护您的财产安全，预防电信网络诈骗。
        
        【系统功能】
        1. 本地监控可疑关键词
        2. 实时预警诈骗风险
        3. 提供反诈教育内容
        4. 协助报警（需您确认）
        
        【隐私保护】
        ✓ 所有数据本地处理，不上传云端
        ✓ 仅监控已知诈骗关键词
        ✓ 不记录正常聊天内容
        ✓ 可随时关闭系统
        
        【法律依据】
        《中华人民共和国反电信网络诈骗法》
        第十七条：公安机关应当会同有关部门建立电信网络诈骗预警劝阻系统。
        
        您是否同意启用龍魂反诈预警系统保护您的安全？
        
        输入：同意 / 不同意
        """
        print(协议)
        选择 = input("您的选择: ").strip()
        self.用户授权 = (选择 == "同意")
        return self.用户授权
    
    def 扫描文本(self, 文本: str) -> Optional[Dict]:
        """
        扫描文本中的诈骗关键词
        仅在本地运行，不上传任何数据
        """
        if not self.用户授权:
            return None
        
        发现的关键词 = []
        最高风险等级 = "低危"
        
        for 诈骗类型, 关键词列表 in self.诈骗关键词库.items():
            for 关键词 in 关键词列表:
                if 关键词 in 文本:
                    风险等级 = self._评估风险等级(关键词)
                    发现的关键词.append({
                        "类型": 诈骗类型,
                        "关键词": 关键词,
                        "等级": 风险等级
                    })
                    
                    if 风险等级 == "高危":
                        最高风险等级 = "高危"
                    elif 风险等级 == "中危" and 最高风险等级 != "高危":
                        最高风险等级 = "中危"
        
        if 发现的关键词:
            return {
                "风险等级": 最高风险等级,
                "发现关键词": 发现的关键词,
                "时间": datetime.now().isoformat(),
                "DNA": self.DNA追溯码
            }
        
        return None
    
    def _评估风险等级(self, 关键词: str) -> str:
        """评估关键词风险等级"""
        高危词 = ["安全账户", "通缉令", "杀猪盘", "涉嫌洗钱"]
        中危词 = ["高回报", "保证金", "解冻费", "垫付"]
        
        if any(词 in 关键词 for 词 in 高危词):
            return "高危"
        elif any(词 in 关键词 for 词 in 中危词):
            return "中危"
        return "低危"
    
    def 触发预警(self, 预警信息: Dict) -> None:
        """
        触发预警，进行思想教育
        不上传任何数据，仅在本地显示
        """
        等级 = 预警信息["风险等级"]
        等级信息 = self.预警等级[等级]
        
        # 显示预警弹窗（模拟）
        弹窗内容 = f"""
        ═══════════════════════════════════════════════════════════
        {等级信息['颜色']} 龍魂反诈预警 - {等级}风险 {等级信息['颜色']}
        ═══════════════════════════════════════════════════════════
        
        ⚠️ 检测到可疑诈骗关键词！
        
        【风险类型】
        """
        
        for 发现 in 预警信息["发现关键词"]:
            弹窗内容 += f"\n  • {发现['类型']} - {发现['关键词']} [{发现['等级']}]"
        
        弹窗内容 += """
        
        【诈骗套路解析】
        """
        
        # 根据类型显示教育内容
        类型列表 = set(发现['类型'] for 发现 in 预警信息["发现关键词"])
        for 类型 in 类型列表:
            弹窗内容 += self._获取教育内容(类型)
        
        弹窗内容 += """
        
        【三个绝不】
        ✗ 绝不向陌生账户转账
        ✗ 绝不透露验证码密码
        ✗ 绝不点击可疑链接
        
        【遇到诈骗怎么办】
        1. 立即停止操作
        2. 拨打110或96110报警
        3. 保存证据（截图、录音）
        4. 通知家人朋友
        
        【是否自愿上报公安机关？】
        （仅上报脱敏关键词，不上传聊天内容）
        
        输入：上报 / 不上报 / 关闭
        ═══════════════════════════════════════════════════════════
        """
        
        print(弹窗内容)
        
        # 记录预警（本地存储）
        self.预警记录.append(预警信息)
    
    def _获取教育内容(self, 诈骗类型: str) -> str:
        """获取针对性教育内容"""
        教育内容库 = {
            "投资诈骗": """
        📊 投资诈骗套路：
           1. 承诺"稳赚不赔"、"高回报"
           2. 要求下载特定APP或平台
           3. 先让你小额投资，再要求大额
           4. 钱被骗走，平台消失
           
           💡 龍魂提醒：
           天上不会掉馅饼，高收益必有高风险！
        """,
            "冒充公检法": """
        👮 冒充公检法套路：
           1. 打电话说你"涉嫌犯罪"
           2. 发送"通缉令""逮捕证"（伪造）
           3. 要求转账到"安全账户"
           4. 威胁"不能告诉家人"
           
           💡 龍魂提醒：
           真警察不会电话办案，更不会让你转账！
        """,
            "刷单诈骗": """
        📱 刷单诈骗套路：
           1. "日赚千元"、"动动手指"
           2. 先让你垫付小额
           3. 要求连单、大额垫付
           4. 钱被骗，无法提现
           
           💡 龍魂提醒：
           刷单违法，先垫钱的都是诈骗！
        """
        }
        return 教育内容库.get(诈骗类型, "\n        ⚠️ 此类诈骗高发，请提高警惕！\n")
    
    def 自愿上报(self, 预警信息: Dict) -> bool:
        """
        用户自愿上报公安机关
        仅上报脱敏数据，不上传原文
        """
        # 脱敏处理：只上报关键词类型，不上报聊天内容
        脱敏数据 = {
            "预警时间": 预警信息["时间"],
            "风险等级": 预警信息["风险等级"],
            "诈骗类型": list(set(发现['类型'] for 发现 in 预警信息["发现关键词"])),
            "关键词哈希": hashlib.sha256(
                json.dumps(预警信息["发现关键词"]).encode()
            ).hexdigest()[:16],
            "上报DNA": self.DNA追溯码
        }
        
        print("\n📤 正在上报反诈中心...")
        print(f"上报内容（已脱敏）: {json.dumps(脱敏数据, ensure_ascii=False)}")
        
        # 这里可以接入公安公开的举报API
        # 但绝不接入涉密内网
        
        print("✅ 上报成功！感谢您的配合，共同打击诈骗！")
        return True
    
    def 生成报告(self) -> str:
        """生成本地反诈报告"""
        报告 = f"""
        ═══════════════════════════════════════════════════════════
                    龍魂反诈预警系统 - 个人防护报告
        ═══════════════════════════════════════════════════════════
        
        报告生成时间: {datetime.now().isoformat()}
        DNA追溯码: {self.DNA追溯码}
        
        【预警统计】
        总预警次数: {len(self.预警记录)}
        
        【风险分布】
        """
        
        风险统计 = {}
        for 记录 in self.预警记录:
            等级 = 记录["风险等级"]
            风险统计[等级] = 风险统计.get(等级, 0) + 1
        
        for 等级, 次数 in 风险统计.items():
            报告 += f"  {self.预警等级[等级]['颜色']} {等级}: {次数}次\n"
        
        报告 += """
        【防护建议】
        1. 安装国家反诈中心APP
        2. 关注"公安部刑侦局"微信公众号
        3. 牢记反诈专线：96110

        【龍魂承诺】

        ═══════════════════════════════════════════════════════════

        return 报告
