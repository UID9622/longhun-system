### 第3步：激活权重计算（策略1）

```bash
# 创建权重计算脚本
cat > ~/.dragonsoul/scripts/calculate_compliance.py << 'EOF'
#!/usr/bin/env python3
"""
龙魂价值观遵守度计算脚本
基于策略1: 权重量化细则
"""

import json
from datetime import datetime

# 龙魂价值观定义
VALUES = {
    "VS001": {"name": "为人民服务", "weight": 0.30},
    "VS002": {"name": "自省进化", "weight": 0.25},
    "VS003": {"name": "数据主权", "weight": 0.20},
    "VS004": {"name": "诚实透明", "weight": 0.15},
    "VS005": {"name": "团结协作", "weight": 0.10},
}

# 示例：计算遵守度
def calculate_compliance():
    # 这里应该是从实际指标中读取数据
    # 现在使用模拟数据
    scores = {
        "VS001": 0.98,  # 为人民服务遵守度
        "VS002": 0.95,  # 自省进化遵守度
        "VS003": 0.90,  # 数据主权遵守度
        "VS004": 1.00,  # 诚实透明遵守度
        "VS005": 0.97,  # 团结协作遵守度
    }

    # 计算总遵守度
    total_compliance = sum(VALUES[v]["weight"] * scores[v] for v in VALUES)

    return {
        "timestamp": datetime.now().isoformat(),
        "compliance": total_compliance,
        "breakdown": {VALUES[v]["name"]: scores[v] for v in VALUES}
    }

if __name__ == "__main__":
    result = calculate_compliance()
    print(json.dumps(result, indent=2, ensure_ascii=False))

    # 保存到审计日志
    with open(f"~/.dragonsoul/audit/compliance_{datetime.now().strftime('%Y%m%d')}.json", 'a') as f:
        f.write(json.dumps(result) + "\n")

EOF

chmod +x ~/.dragonsoul/scripts/calculate_compliance.py
```
