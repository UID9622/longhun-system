### Python徽章生成器

```python
# UID9622 徽章生成器 Python版本
import urllib.parse

class UID9622BadgeGenerator:
    def __init__(self):
        self.base_url = "https://img.shields.io/badge"
        self.colors = {
            'red': 'FF0000',
            'blue': '0066CC',
            'green': '28A745', 
            'purple': '6F42C1',
            'gold': 'FFD700'
        }
    
    def generate_philosophy_badge(self, name, philosophy, color, logo='yinyang'):
        """生成哲学徽章"""
        encoded_name = urllib.parse.quote(name)
        encoded_philosophy = urllib.parse.quote(philosophy)
        return f"![{name}]({self.base_url}-{encoded_name}-{encoded_philosophy}-{self.colors[color]}?style=for-the-badge&logo={logo}&logoColor=white)"
    
    def generate_china_badge(self, name, concept, color='red', logo='dragon'):
        """生成中国主题徽章"""
        encoded_name = urllib.parse.quote(name)
        encoded_concept = urllib.parse.quote(concept)
        return f"![{name}]({self.base_url}-{encoded_name}-{encoded_concept}-{self.colors[color]}?style=for-the-badge&logo={logo}&logoColor=gold)"

# 使用示例
badge_gen = UID9622BadgeGenerator()

print(badge_gen.generate_china_badge("中国原创", "文化自信", "red"))
```
