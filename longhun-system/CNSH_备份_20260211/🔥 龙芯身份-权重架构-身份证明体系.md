# 龍芯DNA追溯码生成器
class DNA追溯码生成器:
    def __init__(self):
        self.基础前缀 = "#龍芯⚡️"
        self.GPG指纹 = "A2D0092CEE2E5BA87035600924C3704A8CC26D5F"
    
    def 生成(self, 操作类型: str, 数据: str) -> str:
        """
        生成唯一的DNA追溯码
        
        参数:
            操作类型: 如"激活","收益","审计"
            数据: 操作相关的数据
            
        返回:
            DNA追溯码
        """
        import hashlib
        import time
        
        时间戳 = time.strftime("%Y-%m-%d")
        数据哈希 = hashlib.sha256(f"{数据}:{self.GPG指纹}".encode()).hexdigest()[:16]
        
        return f"{self.基础前缀}{时间戳}-{操作类型}-{数据哈希}"
    
    def 验证(self, dna码: str, 预期数据: str) -> bool:
        """验证DNA追溯码是否有效"""
        生成码 = self.生成(dna码.split("-")[-2], 预期数据)
        return dna码 == 生成码

### **权重计算的核心公式**

```python
# 龍芯权重算法 - 核心公式

class 龍芯权重计算器:
    def __init__(self):
        self.dna追溯码 = "#龍芯⚡️2026-01-30-权重计算-v1.0"
    
    def 计算权重(self, 身份: Dict[str, Any]) -> float:
        """
        计算用户权重
        
        公式：
        权重 = 基础权重 + 身份加成 + 贡献加成 + 时间加成
        
        参数:
            身份: 包含GPG验证、网络身份、行为历史的字典
        """
        
        # 1. 基础权重（所有人平等）
        基础权重 = 1.0
        
        # 2. 身份加成（GPG验证过的身份）
        身份加成 = 0
        if 身份.get("GPG验证通过"):
            身份加成 += 2.0
        
        # 根据网络身份证类型加成
        网络身份证 = 身份.get("网络身份证", "")
        if 网络身份证.startswith("T38"):  # 军人身份
            身份加成 += 5.0
        elif 网络身份证.startswith("T39"):  # 烈属身份
            身份加成 += 4.0
        elif 网络身份证.startswith("T40"):  # 贡献者
            身份加成 += 3.0
        
        # 3. 贡献加成（帮人行为）
        贡献加成 = 身份.get("帮人次数", 0) * 0.5
        
        # 4. 时间加成（注册时长）
        注册天数 = (datetime.now() - 身份.get("注册时间", datetime.now())).days
        时间加成 = min(注册天数 / 365, 2.0)  # 最多2分
        
        # 总权重
        总权重 = 基础权重 + 身份加成 + 贡献加成 + 时间加成
        
        # 权重封顶（最高10分）
        return min(总权重, 10.0)
    
    def 计算权限等级(self, 权重: float) -> str:
        """根据权重计算权限等级"""
        if 权重 >= 8.0:
            return "先驱者"
        elif 权重 >= 6.0:
            return "守护者"
        elif 权重 >= 4.0:
            return "传承者"
        elif 权重 >= 2.0:
            return "贡献者"
        else:
            return "新手"
```
