## 📋 核心功能

### 1. DNA生成与验证
```python
from dna_verifier import CNSHDNAVerifier

# 创建DNA验证器
verifier = CNSHDNAVerifier()

# 生成DNA标签
dna_info = verifier.generate_dna("AIENGINE")
print(f"DNA标签: {dna_info['dna_code']}")
print(f"LU序列: {dna_info['lu_sequence']}")

# 验证DNA完整性
valid, message = verifier.verify_dna(dna_info['dna_code'])
print(f"验证结果: {valid}, 消息: {message}")
```

### 2. 记忆存储与检索
```python
from dna_memory_system import DNAMemorySystem

# 创建记忆系统
memory_system = DNAMemorySystem()

# 存储记忆
result = memory_system.store_memory(
    content="这是一条重要的对话记录",
    memory_type="DIALOG",
    priority="P1"
)

print(f"记忆存储成功: {result['lu_sequence']}")

# 检索记忆
memory = memory_system.recall_memory(result['lu_sequence'])
if memory:
    print(f"记忆内容: {memory['content']}")
```

## 🎯 使用示例

### 示例1：创建个人对话记录
```python
from dna_memory_system import DNAMemorySystem

memory_system = DNAMemorySystem()

# 存储个人对话
conversation = """
用户：你好，我想了解DNA系统
AI：您好！DNA系统是...
用户：如何使用记忆功能？
AI：记忆功能通过...
"""

result = memory_system.store_memory(conversation, "DIALOG", "P1")
print(f"对话已保存，序列号: {result['lu_sequence']}")
```

### 示例2：批量处理记忆
```python
import os
from dna_memory_system import DNAMemorySystem

memory_system = DNAMemorySystem()

# 批量导入历史数据
history_files = ["历史对话1.txt", "历史对话2.txt"]

for file in history_files:
    if os.path.exists(file):
        with open(file, 'r', encoding='utf-8') as f:
            content = f.read()
            result = memory_system.store_memory(content, "ARCHIVE", "P2")
            print(f"导入完成: {file} -> {result['lu_sequence']}")
```
