## 📝 完整开发环境配置脚本

```bash
#!/bin/bash

# ====================================================
# Mac 一键开发环境配置脚本 (setup_dev_env.sh)
# 版本: 1.0.0
# 创建时间: 2025-08-14
# 作者: 🍀系统中枢｜UID9622
# ====================================================

# 设置日志文件
LOG_FILE="$HOME/setup_log.txt"
echo "=== 开发环境配置开始 $(date) ===" > $LOG_FILE

# 定义颜色输出函数
print_green() {
    echo -e "\033[0;32m$1\033[0m"
    echo "[成功] $1" >> $LOG_FILE
}

print_red() {
    echo -e "\033[0;31m$1\033[0m"
    echo "[错误] $1" >> $LOG_FILE
}

print_yellow() {
    echo -e "\033[0;33m$1\033[0m"
    echo "[警告] $1" >> $LOG_FILE
}

print_blue() {
    echo -e "\033[0;34m$1\033[0m"
    echo "[信息] $1" >> $LOG_FILE
}

# 检查是否安装Homebrew
check_brew() {
    print_blue "检查Homebrew安装状态..."
    if ! command -v brew &> /dev/null; then
        print_yellow "Homebrew未安装，正在安装..."
        /bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
        if [ $? -ne 0 ]; then
            print_red "Homebrew安装失败，请手动安装后重试"
            exit 1
        fi
        print_green "Homebrew安装成功"
    else
        print_green "Homebrew已安装"
        brew update
    fi
}

# 安装与配置Python环境
setup_python() {
    print_blue "开始配置Python环境..."
    
    # 安装Python
    brew install python@3.12
    
    # 安装pip和基本工具
    python3 -m pip install --upgrade pip
    pip3 install virtualenv ipython jupyter pipenv
    
    # 创建基本项目目录
    mkdir -p ~/Projects/Python
    
    # 添加Python路径到.zshrc或.bash_profile
    if [ -f ~/.zshrc ]; then
        echo 'export PATH="/usr/local/opt/python/libexec/bin:$PATH"' >> ~/.zshrc
        source ~/.zshrc
    elif [ -f ~/.bash_profile ]; then
        echo 'export PATH="/usr/local/opt/python/libexec/bin:$PATH"' >> ~/.bash_profile
        source ~/.bash_profile
    fi
    
    if [ $? -eq 0 ]; then
        print_green "Python环境配置完成"
    else
        print_red "Python环境配置失败"
    fi
}

# 安装与配置Node.js环境
setup_nodejs() {
    print_blue "开始配置Node.js环境..."
    
    # 安装NVM (Node Version Manager)
    curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.5/install.sh | bash
    
    # 配置NVM环境
    export NVM_DIR="$HOME/.nvm"
    [ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh"
    
    # 安装最新稳定版Node.js
    nvm install --lts
    nvm use --lts
    
    # 安装常用全局包
    npm install -g yarn npm-check-updates nodemon typescript ts-node
    
    # 创建基本项目目录
    mkdir -p ~/Projects/Node
    
    if [ $? -eq 0 ]; then
        print_green "Node.js环境配置完成"
    else
        print_red "Node.js环境配置失败"
    fi
}

# 配置Git
setup_git() {
    print_blue "开始配置Git环境..."
    
    # 安装Git
    brew install git
    
    # 配置用户信息(需替换为实际信息)
    read -p "输入Git用户名: " git_username
    read -p "输入Git邮箱: " git_email
    
    git config --global user.name "$git_username"
    git config --global user.email "$git_email"
    
    # 配置Git别名
    git config --global alias.co checkout
    git config --global alias.br branch
    git config --global alias.ci commit
    git config --global alias.st status
    
    # 生成SSH密钥
    ssh-keygen -t ed25519 -C "$git_email"
    
    # 输出SSH公钥
    echo "你的SSH公钥:"
    cat ~/.ssh/id_ed25519.pub
    echo "请将上面的SSH密钥添加到你的GitHub/GitLab账户"
    
    if [ $? -eq 0 ]; then
        print_green "Git环境配置完成"
    else
        print_red "Git环境配置失败"
    fi
}

# 安装与配置VS Code
setup_vscode() {
    print_blue "开始配置VS Code..."
    
    # 安装VS Code
    brew install --cask visual-studio-code
    
    # 安装扩展
    code_extensions=(
        "ms-python.python"
        "dbaeumer.vscode-eslint"
        "esbenp.prettier-vscode"
        "ms-azuretools.vscode-docker"
        "redhat.vscode-yaml"
        "golang.go"
        "vscode-icons-team.vscode-icons"
        "github.copilot"
        "ritwickdey.liveserver"
        "ms-vscode.cpptools"
    )
    
    for extension in "${code_extensions[@]}"; do
        code --install-extension $extension
    done
    
    # 配置VS Code设置
    mkdir -p ~/Library/Application\ Support/Code/User/
    cat > ~/Library/Application\ Support/Code/User/settings.json << EOL
{
    "editor.formatOnSave": true,
    "editor.fontFamily": "Fira Code, Menlo, Monaco, 'Courier New', monospace",
    "editor.fontLigatures": true,
    "editor.fontSize": 14,
    "editor.tabSize": 2,
    "editor.minimap.enabled": true,
    "terminal.integrated.fontFamily": "MesloLGS NF",
    "workbench.iconTheme": "vscode-icons",
    "python.linting.enabled": true,
    "python.formatting.provider": "black",
    "files.autoSave": "afterDelay",
    "files.autoSaveDelay": 1000,
    "workbench.colorTheme": "Default Dark+"
}
EOL
    
    if [ $? -eq 0 ]; then
        print_green "VS Code配置完成"
    else
        print_red "VS Code配置失败"
    fi
}

# 安装与配置iTerm2
setup_iterm2() {
    print_blue "开始配置iTerm2..."
    
    # 安装iTerm2
    brew install --cask iterm2
    
    # 安装Oh My Zsh
    sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)" "" --unattended
    
    # 安装Powerlevel10k主题
    git clone --depth=1 https://github.com/romkatv/powerlevel10k.git ${ZSH_CUSTOM:-$HOME/.oh-my-zsh/custom}/themes/powerlevel10k
    
    # 安装常用插件
    git clone https://github.com/zsh-users/zsh-autosuggestions ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-autosuggestions
    git clone https://github.com/zsh-users/zsh-syntax-highlighting.git ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-syntax-highlighting
    
    # 配置.zshrc
    sed -i '' 's/ZSH_THEME="robbyrussell"/ZSH_THEME="powerlevel10k\/powerlevel10k"/g' ~/.zshrc
    sed -i '' 's/plugins=(git)/plugins=(git zsh-autosuggestions zsh-syntax-highlighting)/g' ~/.zshrc
    
    # 添加有用的别名
    cat >> ~/.zshrc << EOL

# 自定义别名
alias ll="ls -la"
alias python="python3"
alias pip="pip3"
alias gs="git status"
alias gc="git commit"
alias gp="git push"
alias gl="git pull"
alias gd="git diff"
alias gco="git checkout"

# 工作目录快捷方式
alias pj="cd ~/Projects"
alias pyp="cd ~/Projects/Python"
alias np="cd ~/Projects/Node"

# 实用函数
function mkcd() {
  mkdir -p "$1" && cd "$1"
}

EOL
    
    if [ $? -eq 0 ]; then
        print_green "iTerm2配置完成"
        print_yellow "请重启终端或运行 'source ~/.zshrc' 使配置生效"
    else
        print_red "iTerm2配置失败"
    fi
}

# 安装Notion
setup_notion() {
    print_blue "开始安装Notion..."
    
    brew install --cask notion
    
    if [ $? -eq 0 ]; then
        print_green "Notion安装完成"
    else
        print_red "Notion安装失败"
    fi
}

# 安装与配置Obsidian
setup_obsidian() {
    print_blue "开始配置Obsidian..."
    
    # 安装Obsidian
    brew install --cask obsidian
    
    # 创建Obsidian笔记目录
    mkdir -p ~/Documents/Obsidian/DevNotes
    
    # 创建一些示例笔记
    mkdir -p ~/Documents/Obsidian/DevNotes/.obsidian
    cat > ~/Documents/Obsidian/DevNotes/开发环境笔记.md << EOL
# 开发环境笔记

## 已安装工具
- Python $(python3 --version)
- Node.js $(node --version)
- Git $(git --version)
- VS Code
- iTerm2
- Notion
- Obsidian

## 常用命令
\`\`\`bash
# Python虚拟环境
python -m venv venv
source venv/bin/activate

# Node.js项目初始化
npm init -y
yarn init -y

# Git基本操作
git init
git add .
git commit -m "Initial commit"
git remote add origin <repo-url>
git push -u origin main
\`\`\`

## 重要链接
- [Python文档](https://docs.python.org/)
- [Node.js文档](https://nodejs.org/docs/latest-v16.x/api/)
- [Git文档](https://git-scm.com/doc)
- [VS Code文档](https://code.visualstudio.com/docs)
EOL
    
    if [ $? -eq 0 ]; then
        print_green "Obsidian配置完成"
        print_blue "在Obsidian中打开文件夹: ~/Documents/Obsidian/DevNotes"
    else
        print_red "Obsidian配置失败"
    fi
}

# 主函数
main() {
    print_blue "=== 开始一键配置开发环境 ==="
    
    # 检查并安装Homebrew
    check_brew
    
    # 配置所有环境
    setup_python
    setup_nodejs
    setup_git
    setup_vscode
    setup_iterm2
    setup_notion
    setup_obsidian
    
    # 记录安装完成时间
    echo "=== 开发环境配置完成 $(date) ===" >> $LOG_FILE
    
    print_green "=== 所有配置已完成！==="
    print_blue "日志文件保存在: $LOG_FILE"
    print_yellow "建议重启电脑以确保所有配置生效"
}

# 执行主函数
main

```

### ❓ 常见问题解答

- 如何在Visual Studio Code中运行此脚本？
	尽管可以在VS Code中打开此脚本文件，但实际运行需要在终端中执行：
	1. 在VS Code中打开脚本文件
	2. 打开VS Code的集成终端（Terminal → New Terminal）
	3. 确保脚本有执行权限：`chmod +x setup_dev_env.sh`
	4. 在终端中执行：`./setup_dev_env.sh`
- 如果我只想安装部分工具怎么办？
	可以修改脚本末尾的main()函数，只保留你需要的模块调用。例如，只安装VS Code和Node.js：
	```bash
	main() {
	    print_blue "=== 开始配置部分开发环境 ==="
	    check_brew
	    setup_nodejs
	    setup_vscode
	    print_green "=== 所选配置已完成！==="
	}
	```
- 配置过程中遇到错误怎么办？
	1. 查看错误信息，通常会提示具体问题
	2. 检查网络连接是否正常
	3. 尝试单独运行出错的模块
	4. 如果Homebrew安装出错，可以尝试手动安装：`/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"`
