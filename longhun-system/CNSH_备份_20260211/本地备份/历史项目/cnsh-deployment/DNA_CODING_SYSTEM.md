## 🛠️ **生成算法**

### 🔧 **Python生成函数**

```python
import hashlib
from datetime import datetime

def generate_dna_code(content_type, region, content, author, version="v1.0"):
    """生成CNSH DNA确认码"""
    year = datetime.now().strftime("%Y")
    
    # 构建基础部分
    base_code = f"ZHUGEXIN⚡️{year}-{content_type}-{region}-{content}-{author}-{version}"
    
    # 生成校验码
    checksum = hashlib.sha256(base_code.encode()).hexdigest()[:4].upper()
    
    # 组合完整DNA码
    full_code = f"#{base_code}-{checksum}"
    
    return full_code

# 使用示例
dna = generate_dna_code("ID", "CN", "匿名投票系统", "USER123", "v1.0")
print(dna)
# 输出: #ZHUGEXIN⚡️2025-ID-CN-匿名投票系统-USER123-v1.0-A3F2
```

### 🎯 **验证函数**

```python
def verify_dna_code(dna_code):
    """验证DNA码的有效性"""
    try:
        # 分解DNA码
        parts = dna_code.split('-')
        if len(parts) < 5:
            return False, "格式错误"
        
        # 提取校验码
        provided_checksum = parts[-1]
        base_without_checksum = '-'.join(parts[:-1])
        
        # 重新计算校验码
        expected_checksum = hashlib.sha256(base_without_checksum.encode()).hexdigest()[:4].upper()
        
        # 验证
        return provided_checksum == expected_checksum, "校验成功" if provided_checksum == expected_checksum else "校验失败"
    
    except Exception as e:
        return False, f"验证失败: {str(e)}"

# 使用示例
valid, message = verify_dna_code("#ZHUGEXIN⚡️2025-ID-CN-匿名投票系统-USER123-v1.0-A3F2")
print(valid, message)  # 输出: True 校验成功
```
