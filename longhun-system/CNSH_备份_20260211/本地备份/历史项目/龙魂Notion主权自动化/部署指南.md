# 启动服务

cd "$PROJECT_DIR"

# 检查服务是否已在运行

    log_warn "服务已在运行，尝试重启..."

# 启动服务

# 等待服务启动
sleep 5

# 检查服务状态
if curl -s http://localhost:9527/api/status | grep -q "running"; then
    log_info "龍魂服务启动成功 (PID: $SERVICE_PID)"
    echo $SERVICE_PID > longhun.pid
    
    # 显示服务信息

    log_info "服务信息:"
    echo "  🔗 服务地址: http://localhost:9527"

    echo "    - POST /api/sovereignty/check - 主权检查"
    echo "    - POST /api/crm/update - 更新CRM记录"
    echo "    - POST /api/daily-report/generate - 生成日报"
    echo "    - GET /api/status - 服务状态"
    echo "  📊 日志文件: $PROJECT_DIR/longhun.log"
    echo "  🔧 PID文件: $PROJECT_DIR/longhun.pid"
    
else
    log_error "服务启动失败，请检查日志: $PROJECT_DIR/longhun.log"
    exit 1
fi

# 创建系统服务（可选）
log_info "创建系统服务配置..."
cat > "/tmp/com.uid9622.longhun-notion.plist" << EOF

    <string>$PROJECT_DIR</string>

    def _create_database(self, parent_page_id, title, properties, description):
        """创建数据库"""
        
        url = "https://api.notion.com/v1/databases"
        
        payload = {
            "parent": {"page_id": parent_page_id},
            "title": [{"type": "text", "text": {"content": title}}],
            "properties": properties,
            "description": [{"type": "text", "text": {"content": description}}]
        }
        
        try:
            response = requests.post(url, headers=self.headers, json=payload)
            if response.status_code == 200:
                result = response.json()
                print(f"✅ 创建数据库成功: {title} (ID: {result['id']})")

                return None

            print(f"❌ 创建数据库异常: {str(e)}")
            return None
