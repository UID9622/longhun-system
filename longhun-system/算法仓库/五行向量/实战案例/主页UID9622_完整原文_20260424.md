class PersonaShadow:
    def __init__(self, main_persona, scenario, constraints):
        self.main = main_persona  # 主人格
        self.scenario = scenario  # 历史场景或假设情境
        self.constraints = constraints  # 约束条件（信息限制、资源限制等）
        self.memory_fork = main_persona.memory.fork()  # 记忆分叉
        
    def simulate(self, decision_chain):
        """模拟决策链的结果"""
        results = []
        for decision in decision_chain:
            outcome = self.scenario.execute(decision, self.constraints)
            results.append({
                "decision": decision,
                "outcome": outcome,
                "alignment_score": self.check_alignment(outcome)
            })
        return results
    
    def check_alignment(self, outcome):
        """检查结果是否符合龙魂价值观"""
        return DragonSoulValues.evaluate(outcome)
    
    def merge_learnings(self):
        """将试炼中的经验合并回主人格"""
        learnings = self.extract_patterns()
        self.main.knowledge_base.append(learnings)
        return learnings

class HistoricalScenario:
    def __init__(self, time_point, context):
        self.time_point = time_point  # "2008金融危机前夕"
        self.context = context  # 历史背景数据
        self.timeline = Timeline(time_point)
        
    def execute(self, decision, constraints):
        """在约束条件下执行决策，返回结果"""
        delayed_info = self.apply_info_delay(constraints.get("info_delay"))
        capital_limit = constraints.get("capital")
        
        outcome = self.timeline.forward(
            action=decision,
            available_info=delayed_info,
            resources=capital_limit
        )
        return outcome
    
    def apply_info_delay(self, delay_period):
        """模拟信息延迟（如6个月前的数据）"""
        return self.context.get_historical_data(delay=delay_period)

class DragonSoulValues:
    CORE_PRINCIPLES = {
        "智慧": ["长期思维", "系统性思考", "风险管理"],
        "仁义": ["保护团队", "诚信透明", "共同成长"],
        "勇气": ["果断决策", "承担责任", "创新突破"],
        "平衡": ["阴阳调和", "刚柔并济", "动态适应"]
    }
    
    @staticmethod
    def evaluate(outcome):
        """评估结果与龙魂价值观的契合度"""
        score = 0
        for principle, criteria in DragonSoulValues.CORE_PRINCIPLES.items():
            score += outcome.match_score(criteria)
        return score / len(DragonSoulValues.CORE_PRINCIPLES)

class ExperienceMerger:
    def __init__(self, shadow_results):
        self.results = shadow_results
        
    def extract_patterns(self):
        """提取决策模式和教训"""
        patterns = {
            "成功模式": [],
            "失败教训": [],
            "临界点识别": [],
            "情境依赖性": []
        }
        
        for result in self.results:
            if result["alignment_score"] > 0.7:
                patterns["成功模式"].append(result)
            else:
                patterns["失败教训"].append(result)
                
            # 识别关键决策节点
            if result["outcome"].is_critical:
                patterns["临界点识别"].append(result)
                
        return patterns
    
    def integrate_to_main(self, main_persona):
        """整合到主人格的决策树中"""
        patterns = self.extract_patterns()
        
        # 更新决策权重

            main_persona.decision_weights.increase(
                success["decision"],

            )
            
        # 添加风险警示
        for failure in patterns["失败教训"]:
            main_persona.risk_alerts.add(
                failure["decision"],
                warning=failure["outcome"].lessons
            )
            
        return main_persona

class MultiShadowSimulator:

        self.main = main_persona
        self.shadows = []
        
    def spawn_shadows(self, scenarios):
        """在多个平行场景中创建分身"""
        for scenario in scenarios:
            shadow = PersonaShadow(
                main_persona=self.main,
                scenario=scenario,
                constraints=scenario.constraints
            )
            self.shadows.append(shadow)
            
    def parallel_simulate(self):
        """并行执行所有分身的模拟"""
        all_results = []
        for shadow in self.shadows:
            results = shadow.simulate(shadow.scenario.decision_chain)
            all_results.append(results)
        return all_results
    
    def synthesize_learnings(self):
        """综合所有分身的经验"""
        meta_patterns = {}
        for shadow in self.shadows:
            learnings = shadow.merge_learnings()

  ## 0️⃣ 理论起点（1943–1956）
  ### AI-1943-01｜1943 — McCulloch–Pitts 人工神经元
  🔬 阶段： Theory　标签： #AI #NeuralNetwork #Theory
  核心算法
  神经元 = 加权求和 + 阈值判断。多个输入信号，每个乘上重要程度（权重），加起来，超过门槛就输出1，否则输出0。这是"神经元"从生物概念变成数学公式的第一步。
  关键变量
  - $x_i$：第 i 个输入信号（来自其他神经元或外部）
  - $w_i$：第 i 个权重（这个信号的"重要程度"）
  - $theta$：激活阈值（加权和超过它才触发）
  - $y$：输出（0或1）
  技术语法/公式
  y = \begin{cases} 1 & \text{if } \sum_i w_i x_i \geq \theta \\ 0 & \text{otherwise} \end{cases}
  工程意义
  神经网络从"概念"变成"可以用电路实现的逻辑"，奠定了所有后续神经网络的最底层基础。

  ### AI-1958-01｜1958 — 感知机（Perceptron）
  🔬 阶段： Theory　标签： #AI #Perceptron #Learning
  核心算法
  Rosenblatt的感知机：让机器通过错误来修正权重。预测错了就调整，调整方向是"让下次不错"。这是机器学习中"从错误中学习"的最早工程实现。
  关键变量
  - $w$：权重向量（机器需要学习的参数）
  - $eta$：学习率（每次调整的步长大小）
  - $y$：预测输出
  - $hat{y}$：真实标签
  技术语法/公式
  w \leftarrow w + \eta (\hat{y} - y) x
  工程意义
  第一个可以"自动学习"的分类器，权重更新法则是梯度下降的前身。

  ## 1️⃣ 统计学习时代（1980–2005）
  ### AI-1986-01｜1986 — 反向传播算法（Backpropagation）
  📊 阶段： Statistical　标签： #AI #Backprop #DeepLearning
  核心算法
  链式法则+梯度下降的联姻。从输出层的误差出发，逐层往回算每个权重对误差的贡献（梯度），然后按贡献大小同步更新。让多层网络第一次可以被训练。
  关键变量
  - $L$：损失函数（预测值和真实值的差距）
  - $w$：需要更新的权重
  - $eta$：学习率
  - $frac{partial L}{partial w}$：损失对权重的梯度
  技术语法/公式
  w \leftarrow w - \eta \cdot \frac{\partial L}{\partial w}
  链式求导核心：
  \frac{\partial L}{\partial w_1} = \frac{\partial L}{\partial z_2} \cdot \frac{\partial z_2}{\partial z_1} \cdot \frac{\partial z_1}{\partial w_1}
  工程意义
  神经网络从"概念"变"可迭代训练"，是深度学习所有训练算法的基础，今天还在用。

  ### AI-1995-01｜1995 — 支持向量机（SVM）
  📊 阶段： Statistical　标签： #AI #SVM #Classification
  核心算法
  找一个"最胖的"决策边界：两类数据中间，找离两边都最远的超平面。边界越宽，泛化越好。核技巧（Kernel Trick）让它能处理非线性问题。
  关键变量
  - $x_i$：输入样本特征向量
  - $y_i$：标签（+1或-1）
  - $w$：超平面法向量
  - $b$：偏置
  - $alpha_i$：拉格朗日乘子（支持向量的系数）
  技术语法/公式
  决策函数：
  f(x) = \text{sign}(w^T x + b)
  最大间隔优化目标：
  \min_{w,b} \frac{1}{2}\|w\|^2 \quad \text{s.t.} \quad y_i(w^T x_i + b) \geq 1
  工程意义
  深度学习崛起前最主流的分类器，理论保证强，至今仍在小数据场景广泛应用。
