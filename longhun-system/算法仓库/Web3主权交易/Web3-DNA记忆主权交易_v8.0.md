### 2.3 函数 ① 计算五行强度（四柱权重）

```python
位置权重表 = {
    "年柱": {"天干":1.0,"地支":0.8},
    "月柱": {"天干":1.5,"地支":1.2},
    "日柱": {"天干":2.0,"地支":1.6},   # 本命主柱·权重最高
    "时柱": {"天干":1.2,"地支":1.0},
}

def 计算五行强度(四柱):
    得分 = {k:0.0 for k in 五行}
    for 柱, 干支 in 四柱.items():
        w = 位置权重表[柱]
        if 干支["天干"] in 天干五行表:
            得分[天干五行表[干支["天干"]]] += w["天干"]
        if 干支["地支"] in 地支五行表:
            得分[地支五行表[干支["地支"]]] += w["地支"]
    总分 = sum(得分.values())
    均值 = 总分 / 5
    方差 = sum((v-均值)**2 for v in 得分.values()) / 5
    均衡指数 = max(0.0, round(1.0 - (方差**0.5)/(均值+0.001), 3))
    return {
        "五行得分": 得分,
        "最强": max(得分, key=得分.get),
        "最弱": min(得分, key=得分.get),
        "均衡指数": 均衡指数,
        "缺失五行": [k for k,v in 得分.items() if v == 0.0]
    }
```

### 2.4 函数 ② 完整链路分析（断链 / 过旺）

```python
def 完整链路分析(得分):
    相生顺序 = ["金","水","木","火","土"]
    断链预警 = []
    健康度 = 100
    for i in range(5):
        来源 = 相生顺序[i]
        目标 = 相生顺序[(i+1) % 5]
        if 得分.get(来源,0) > 0 and 得分.get(目标,0) == 0:
            断链预警.append(f"🔴 断链：{来源}({得分[来源]:.1f}) 生不出 {目标}(0)")
            健康度 -= 15
    总分 = sum(得分.values()) + 0.001
    for k, v in 得分.items():
        if v / 总分 > 0.40:
            疏导 = 五行相生[k]
            断链预警.append(f"🟡 过旺：{k} 占 {v/总分:.0%}·疏导→{疏导}")
            健康度 -= 10
    return {
        "链路健康度": max(0, 健康度),
        "状态": "🟢 健康" if 健康度>=80 else "🟡 待关注" if 健康度>=50 else "🔴 需干预",
        "断链预警": 断链预警
    }
```

def 计算数字根(文本):
    数字 = [int(c) for c in 文本 if c.isdigit()]
    if not 数字: return 0
    s = sum(数字)
    while s >= 10:
        s = sum(int(c) for c in str(s))
    return s
