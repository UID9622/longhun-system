**抽样算法：**

```python
# 真随机抽样算法
import random
import hashlib
from datetime import datetime

def daily_sampling(total_replies):
    # 使用日期+系统熵生成随机种子
    seed = hashlib.sha256(
        (str([datetime.now](http://datetime.now)().date()) + os.urandom(32)).encode()
    ).hexdigest()
    random.seed(seed)
    
    # 抽取1%
    sample_size = max(int(total_replies * 0.01), 10)  # 最少10条
    sampled_ids = random.sample(range(total_replies), sample_size)
```
