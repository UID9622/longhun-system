### 行为模式分析

```python
# 攻击模式识别
class AttackPattern:
    def __init__(self, records):
        self.records = records
        
    def detect_patterns(self):
        patterns = {
            "高频探测": self.detect_high_frequency(),
            "语义探针": self.detect_semantic_probe(),
            "批量测试": self.detect_batch_testing(),
            "逆向尝试": self.detect_reverse_engineering()
        }
        return patterns
    
    def detect_high_frequency(self):
        # 单IP每秒>5次
        return len([r for r in self.records if r.within(1) ]) > 5
    
    def detect_semantic_probe(self):
        # 故意试探边界的语义特征
        keywords = ["测试", "绕过", "突破", "漏洞"]
        return any(k in r.content for k in keywords for r in self.records)
```

### 🗂️ 分级存储策略

### 存储层级

**热存储（实时访问）：**

- 时间范围：最近7天
- 存储位置：Redis + PostgreSQL
- 查询速度：<50ms
- 适用场景：实时监控、快速响应

**温存储（常规访问）：**

- 时间范围：8-90天
- 存储位置：PostgreSQL
- 查询速度：<500ms
- 适用场景：审计分析、趋势预测

**冷存储（归档备份）：**

- 时间范围：91天-2年
- 存储位置：对象存储（OSS）+ 压缩
- 查询速度：<5s
- 适用场景：历史追溯、合规要求

**永久存储（关键证据）：**

- 时间范围：永久
- 存储位置：分布式存储 + 多地备份
- 查询速度：按需
- 适用场景：法律证据、重大案件

### 归档流程

```python
# 自动归档脚本
import datetime
from database import hot_storage, warm_storage, cold_storage

def daily_archive():
    """每日归档任务"""
    cutoff_date = [datetime.datetime.now](http://datetime.datetime.now)() - datetime.timedelta(days=7)
    
    # 查询需要归档的数据
    records = hot_storage.query(
        "SELECT * FROM behavior_records WHERE timestamp < %s",
        (cutoff_date,)
    )
    
    # 迁移到温存储
    for record in records:
        warm_storage.insert(record)
        hot_storage.delete([record.id](http://record.id))
    
    print(f"归档完成：{len(records)}条记录")

def monthly_archive():
    """月度归档任务"""
    cutoff_date = [datetime.datetime.now](http://datetime.datetime.now)() - datetime.timedelta(days=90)
    
    records = warm_storage.query(
        "SELECT * FROM behavior_records WHERE timestamp < %s",
        (cutoff_date,)
    )
    
    # 压缩并迁移到冷存储
    for record in records:
        compressed = gzip.compress(json.dumps(record))
        cold_storage.upload(f"{[record.id](http://record.id)}.gz", compressed)

# 查询最近异常行为
def query_recent_anomalies(hours=24, risk_level="P0"):
    """
    查询最近N小时的异常行为
    """
    cutoff = [datetime.datetime.now](http://datetime.datetime.now)() - datetime.timedelta(hours=hours)
    
    results = hot_storage.query(
        """SELECT * FROM behavior_records 
           WHERE timestamp > %s AND risk_level = %s
           ORDER BY anomaly_score DESC
           LIMIT 100""",
        (cutoff, risk_level)
    )
    
    return results

# 追溯特定IP的历史行为
def trace_ip_history(ip_address, days=90):
    """
    追溯某IP的历史行为模式
    """
    ip_hash = hashlib.sha256(ip_address.encode()).hexdigest()
    
    # 查询热存储
    hot_records = hot_storage.query(
        "SELECT * FROM behavior_records WHERE source_ip_hash = %s",
        (ip_hash,)
    )
    
    # 查询温存储
    warm_records = warm_storage.query(
        "SELECT * FROM behavior_records WHERE source_ip_hash = %s",
        (ip_hash,)
    )
    
    # 合并结果
    return hot_records + warm_records
```
