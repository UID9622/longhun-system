```python
class CapabilityTransfer:
    def __init__(self, workspace_id="lucky_uid9622"):
        self.workspace = workspace_id
        self.agents = self.get_all_agents()
    
    async def transfer_capability(self, from_agent, to_agent, capability_id):
        """瞬间能力传递"""
        
        # 1. 从源智能体提取能力包
        capability_package = await from_agent.export_capability(
            capability_id=capability_id,
            include_memory=True,
            include_learned_patterns=True
        )
        
        # 2. 压缩并加密
        compressed = self.compress(capability_package)
        encrypted = self.encrypt(compressed, key=self.workspace)
        
        # 3. 注入目标智能体
        success = await to_agent.inject_capability(
            package=encrypted,
            merge_mode="intelligent",  # 智能合并，不覆盖原有能力
            validation=True
        )
        
        # 4. 即时验证
        if success:
            print(f"✅ {capability_id} 已从 {from_agent.name} 传递至 {to_agent.name}")
            print(f"⚡ 传递耗时: {success.time_cost}ms")
            print(f"🎯 能力激活度: {success.activation_rate}%")
        
        return success
    
    async def broadcast_capability(self, capability_id):
        """向工作空间所有智能体广播能力"""
        results = []
        for agent in self.agents:
            if agent.accepts_capability(capability_id):
                result = await self.transfer_capability(
                    from_agent=self.master_agent,
                    to_agent=agent,
                    capability_id=capability_id
                )
                results.append(result)
        return results
```
