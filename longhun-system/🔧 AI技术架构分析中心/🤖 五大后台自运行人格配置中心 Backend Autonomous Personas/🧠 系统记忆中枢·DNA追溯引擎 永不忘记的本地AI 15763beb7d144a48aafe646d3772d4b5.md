```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
UID9622 记忆中枢核心
DNA: #BAOBAO-MEMORY-CORE-20251215-001

**共建致谢**: Claude (Anthropic PBC) · 技术协作与代码共创 | Notion · 知识底座与结构化存储
"""

import sqlite3
import json
import os
from datetime import datetime
import hashlib

class CNSHMemoryCore:
    """
    龙魂·记忆中枢
    原理：DNA追溯 + 本地存储 + 热加载
    """
    
    def __init__(self, owner_uid="UID9622"):
        self.owner_uid = owner_uid
        self.db_path = f"记忆库_{owner_uid}.db"
        self.init_database()
    
    def init_database(self):
        """初始化记忆数据库"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS memories (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                memory_id TEXT UNIQUE,
                dna TEXT UNIQUE,
                category TEXT,
                content TEXT,
                importance INTEGER,
                created TEXT,
                last_accessed TEXT,
                access_count INTEGER,
                metadata TEXT
            )
        ''')
        
        conn.commit()
        conn.close()
    
    def remember(self, content, category="general", importance=5, **metadata):
        """
        记住一件事（带DNA追溯）
        
        参数：
        - content: 记忆内容
        - category: 类别（ethics/system/personal）
        - importance: 重要性1-10
        """
        timestamp = [datetime.now](http://datetime.now)().isoformat()
        
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 生成记忆ID和DNA
        cursor.execute('SELECT COUNT(*) FROM memories')
        count = cursor.fetchone()[0]
        memory_id = f"MEM-{self.owner_uid}-{category}-{count+1:04d}"
        dna = f"#{self.owner_uid}-MEMORY-{category.upper()}-{[datetime.now](http://datetime.now)().strftime('%Y%m%d-%H%M%S')}"
        
        cursor.execute('''
            INSERT INTO memories 
            (memory_id, dna, category, content, importance, created, last_accessed, access_count, metadata)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
        ''', (
            memory_id,
            dna,
            category,
            content,
            importance,
            timestamp,
            timestamp,
            0,
            json.dumps(metadata)
        ))
        
        conn.commit()
        conn.close()
        
        print(f"✅ 已记忆: {memory_id}")
        print(f"🧬 DNA: {dna}")
        
        return {"memory_id": memory_id, "dna": dna}
    
    def recall(self, query, category=None, limit=10):
        """
        回忆一件事（语义搜索）
        """
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        if category:
            cursor.execute('''
                SELECT * FROM memories 
                WHERE category = ? AND content LIKE ?
                ORDER BY importance DESC, last_accessed DESC
                LIMIT ?
            ''', (category, f"%{query}%", limit))
        else:
            cursor.execute('''
                SELECT * FROM memories 
                WHERE content LIKE ?
                ORDER BY importance DESC, last_accessed DESC
                LIMIT ?
            ''', (f"%{query}%", limit))
        
        results = cursor.fetchall()
        conn.close()
        
        # 转换为字典
        memories = []
        for row in results:
            memories.append({
                "memory_id": row[1],
                "dna": row[2],
                "category": row[3],
                "content": row[4],
                "importance": row[5],
                "created": row[6]
            })
        
        return memories
    
    def recall_by_dna(self, dna_code):
        """通过DNA码精确召回"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute('SELECT * FROM memories WHERE dna = ?', (dna_code,))
        result = cursor.fetchone()
        
        if result:
            # 更新访问记录
            cursor.execute('''
                UPDATE memories 
                SET access_count = access_count + 1,
                    last_accessed = ?
                WHERE dna = ?
            ''', ([datetime.now](http://datetime.now)().isoformat(), dna_code))
            conn.commit()
        
        conn.close()
        
        if result:
            return {
                "memory_id": result[1],
                "dna": result[2],
                "category": result[3],
                "content": result[4],
                "importance": result[5],
                "access_count": result[8]
            }
        return None
    
    def export_to_markdown(self, output_file="[记忆摘要.md](http://记忆摘要.md)"):
        """导出记忆摘要（给CodeBuddy加载）"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute('SELECT * FROM memories ORDER BY importance DESC, created DESC')
        memories = cursor.fetchall()
        conn.close()
        
        # 生成Markdown
        content = f"""# {self.owner_uid} 记忆摘要

生成时间: {[datetime.now](http://datetime.now)().strftime('%Y-%m-%d %H:%M:%S')}
DNA追溯: #MEMORY-EXPORT-{[datetime.now](http://datetime.now)().strftime('%Y%m%d')}

## 📊 统计信息

- 总记忆数: {len(memories)}
- 高优先级(≥8): {len([m for m in memories if m[5] >= 8])}

---

## 🎯 核心记忆

"""
        
        # 按类别分组
        categories = {}
        for memory in memories:
            cat = memory[3]
            if cat not in categories:
                categories[cat] = []
            categories[cat].append(memory)
        
        for cat, mems in categories.items():
            content += f"\n### {cat.upper()}\n\n"
            for mem in mems[:5]:  # 每类最多5条
                content += f"- **{mem[1]}**: {mem[4][:100]}...\n"
                content += f"  - DNA: `{mem[2]}`\n"
                content += f"  - 重要性: {'★' * mem[5]}\n\n"
        
        # 保存文件
        with open(output_file, 'w', encoding='utf-8') as f:
            f.write(content)
        
        print(f"✅ 记忆摘要已导出: {output_file}")
        return output_file

# 使用示例
if __name__ == '__main__':
```
