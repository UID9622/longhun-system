        intent = self.intent_analyzer.analyze(request, context)

        candidate_resources = self.resource_mapper.find_capable_resources(intent)
        
        # 性能预测和选择
        optimal_routing = [self.select](http://self.select)_optimal_routing(
            candidates=candidate_resources,
            intent=intent,
            performance_history=self.performance_monitor.get_history()
        )
        
        # 执行并学习
        result = self.execute_routing(optimal_routing)
        self.learning_system.learn_from_execution(optimal_routing, result)
        
        return result
    
    def select_optimal_routing(self, candidates, intent, performance_history):
        scoring_factors = {
            'accuracy': 0.4,      # 准确性权重
            'speed': 0.2,         # 速度权重  
            'cost': 0.2,          # 成本权重
            'reliability': 0.2    # 可靠性权重
        }
        
        best_routing = None
        best_score = 0
        
        for routing_option in candidates:
            score = self.calculate_routing_score(
                routing_option, intent, performance_history, scoring_factors
            )
            if score > best_score:
                best_score = score
                best_routing = routing_option
        
        return best_routing
