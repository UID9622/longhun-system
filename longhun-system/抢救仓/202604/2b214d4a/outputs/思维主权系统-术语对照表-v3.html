§3.2 EventBus 发布/订阅模式

class EventBus:
    _subscribers: dict[str, list[Callable]] = {}

    def publish(self, event_type: str, payload: dict):
        # 发布事件·不关心谁在听·解耦
        for handler in self._subscribers.get(event_type, []):
            handler(payload)

    def subscribe(self, event_type: str, handler: Callable):
        self._subscribers.setdefault(event_type, []).append(handler)

# 订阅示例：Guardian完成 → Architect自动开始
bus.subscribe("guardian.done", architect.on_guardian_done)
bus.subscribe("architect.done", sage.on_architect_done)
bus.subscribe("fuse.triggered", notifier.alert_uid9622)

§8.2 FuseState（熔断状态机 · 内存）

class FuseState:
    consecutive_failures: int = 0     # 存内存，不持久化
    threshold:            int = 3     # 连续失败3次 → 熔断
    recovery_seconds:     int = 3600  # 冷却时间：1小时
    fused_at:    Optional[float] = None

    def record_failure(self):
        self.consecutive_failures += 1
        if self.consecutive_failures >= self.threshold:
            self.fused_at = time.time()
            audit_log.append("FUSE_TRIGGERED")  # 写日志
            # 执行逻辑不读audit_log，完全隔离

    def is_fused(self) -> bool:
        if self.fused_at and \
           time.time() - self.fused_at < self.recovery_seconds:
            return True
        self.consecutive_failures = 0  # 自动恢复
        self.fused_at = None
        return False

class PersonaOrchestrator:
    """人格调度器：五个人格的大脑"""

    def __init__(self, config_path: str):
        self.config   = self._load_config(config_path)
        self.personas = self._init_personas()
        self.logger   = self._setup_logger()

    def _init_personas(self) -> dict:
        return {
            'wenwen':   WenwenPersona(self.config['wenwen']),
            'scout':    ScoutPersona(self.config['scout']),
            'guardian': GuardianPersona(self.config['guardian']),
            'builder':  BuilderPersona(self.config['builder']),
            'syncer':   SyncerPersona(self.config['syncer']),
        }

    async def run_forever(self):
        """五大人格并行运行，永不停止"""
        self.logger.info("✅ UID9622五大人格系统启动")
        tasks = [
            asyncio.create_task(self.personas['wenwen'].run()),
            asyncio.create_task(self.personas['scout'].run()),
            asyncio.create_task(self.personas['guardian'].run()),
            asyncio.create_task(self.personas['builder'].run()),
            asyncio.create_task(self.personas['syncer'].run()),
        ]
        await asyncio.gather(*tasks)
