### 场景识别算法（JavaScript移植版）

```javascript
function identifyDDJScene(userInput) {
  let bestScene = null, bestScore = 0;

  for (const [sceneKey, scene] of Object.entries(DDJ_SCENES)) {
    let score = 0;
    for (const kw of scene.keywords) {
      if (userInput.includes(kw)) score += 1;
    }
    if (score > bestScore) {
      bestScore = score;
      bestScene = sceneKey;
    }
  }

  // 情绪关键词检测
  const emotionMap = {
    '🔴': ['完了', '崩', '绝望', '没意义'],
    '🟡': ['纠结', '迷茫', '着急', '困惑'],
    '🟢': ['好奇', '规划', '布局']
  };
  let detectedEmotion = '🟡'; // default
  for (const [em, words] of Object.entries(emotionMap)) {
    if (words.some(w => userInput.includes(w))) { detectedEmotion = em; break; }
  }

  // 紧急度
  const urgencyMap = {10: ['急', '马上', '立即'], 7: ['尽快', '赶紧'], 3: ['慢慢', '有空']};
  let urgency = bestScene ? DDJ_SCENES[bestScene].urgency_base : 5;
  for (const [u, words] of Object.entries(urgencyMap)) {
    if (words.some(w => userInput.includes(w))) { urgency = parseInt(u); break; }
  }

  return {
    scene: bestScene,
    sceneName: bestScene ? DDJ_SCENES[bestScene].name : null,
    emotion: detectedEmotion,
    urgency,
    primaryChapter: bestScene ? DDJ_SCENES[bestScene].primary : null,
    confidence: bestScene ? Math.min(0.95, 0.6 + bestScore * 0.15) : 0
  };
}
```
