```python
from flask import Flask, request, jsonify
import storage
import companion_core

app = Flask(__name__)

# MCP推理接口（原有）
@app.route('/infer', methods=['POST'])

# 情感陪伴相关接口

@app.route('/auth/enable', methods=['POST'])
def enable_companion():
    """启用情感陪伴功能"""
    data = request.get_json(force=True)
    user_id = data.get('user_id')
    level = data.get('level', 1)
    agreement = data.get('agreement', False)
    
    if not agreement:
        return jsonify({'error': '必须同意授权协议'}), 400
    
    if level == 3:
        age_confirm = data.get('age_18_plus', False)
        if not age_confirm:
            return jsonify({'error': '虚拟伴侣功能仅限18岁以上'}), 400
    
    storage.update_user_auth(user_id, level)
    
    return jsonify({
        'success': True,
        'level': level,
        'message': f'已启用第{level}级情感陪伴功能'
    })

@app.route('/auth/status', methods=['GET'])
def get_auth_status():
    """获取用户授权状态"""
    user_id = request.args.get('user_id')
    if not user_id:
        return jsonify({'error': '缺少user_id参数'}), 400
    
    auth_info = storage.get_user_auth_info(user_id)
    return jsonify(auth_info)

@app.route('/infer-companion', methods=['POST'])
def infer_companion():
    """情感陪伴推理接口"""
    data = request.get_json(force=True)
    user_id = data.get('user_id')
    text = data.get('text', '').strip()
    
    if not text:
        return jsonify({'error': 'text 不能为空'}), 400
    
    # 检查授权等级
    auth_level = storage.get_user_auth(user_id)
    if auth_level == 0:
        return jsonify({
            'error': '请先启用情感陪伴功能',
            'hint': '使用 POST /auth/enable 启用'
        }), 403
    
    # 识别情绪
    emotions = companion_core.detect_emotion(text)
    
    # 调度陪伴人格
    companion = companion_core.schedule_companion_persona(emotions, auth_level)
    
    # 生成回复
    if companion == "xinling":
        response = companion_core.generate_xinling_response(text, emotions)
    elif companion == "zhiji":
        response = companion_core.generate_zhiji_response(text, emotions)
    else:
        response = "我在这里陪着你。"
    
    # 写入审计
    storage.write_companion_audit(
        user_id, text, emotions, companion, response, auth_level
    )
    
    return jsonify({
        'emotions': emotions,
        'companion': companion,
        'response': response,
        'auth_level': auth_level
    })

@app.route('/auth/activate', methods=['POST'])
def activate_subscription():
    """激活订阅"""
    data = request.get_json(force=True)
    user_id = data.get('user_id')
    activation_code = data.get('activation_code')
    
    if not user_id or not activation_code:
        return jsonify({'error': '缺少必要参数'}), 400
    
    success, message = activation.activate_user(user_id, activation_code)
    
    if success:
        return jsonify({'success': True, 'message': message})
    else:
        return jsonify({'error': message}), 400

@app.route('/companion-audit', methods=['GET'])
def get_companion_audit():
    """获取陪伴审计记录（管理员）"""
    # 管理员权限检查

    if admin_key != 'CNSH-ADMIN-2025':
        return jsonify({'error': '无权限'}), 403
    
    user_id = request.args.get('user_id')
    limit = int(request.args.get('limit', 50))
    
    records = storage.get_companion_audit_records(user_id, limit)
    return jsonify({'records': records})
```

### 2. storage.py - 基础数据存储

**功能职责：**
- SQLite数据库初始化
- 基础CRUD操作
- 数据库连接管理

**核心表结构：**

```python
import sqlite3
import json
import datetime
from pathlib import Path

DB_PATH = "mcp_flask.db"

def init_database():
    """初始化数据库"""
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    
    # 用户授权表
    c.execute("""
        CREATE TABLE IF NOT EXISTS user_auth (
            user_id TEXT PRIMARY KEY,
            auth_level INTEGER DEFAULT 0,
            auth_time TEXT,
            last_interaction TEXT,
            emotion_state TEXT,
            companion_enabled BOOLEAN DEFAULT 0,
            virtual_partner_enabled BOOLEAN DEFAULT 0,
            created_at TEXT DEFAULT CURRENT_TIMESTAMP
        )
    """)
    
    # 用户订阅表
    c.execute("""
        CREATE TABLE IF NOT EXISTS user_subscription (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id TEXT,
            level INTEGER,
            activation_code TEXT,
            expire_time TEXT,
            activated_time TEXT,
            status TEXT DEFAULT 'active',
            created_at TEXT DEFAULT CURRENT_TIMESTAMP
        )
    """)
    
    # 陪伴审计表
    c.execute("""
        CREATE TABLE IF NOT EXISTS companion_audit (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id TEXT,
            input_text TEXT,
            emotions TEXT,  -- JSON格式
            companion_persona TEXT,
            response_text TEXT,
            auth_level INTEGER,
            timestamp TEXT DEFAULT CURRENT_TIMESTAMP,
            audit_status TEXT DEFAULT 'approved'
        )
    """)
    
    # 对话会话表（防沉迷）
    c.execute("""
        CREATE TABLE IF NOT EXISTS conversation_sessions (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id TEXT,
            start_time TEXT,
            end_time TEXT,
            duration_seconds INTEGER,
            message_count INTEGER DEFAULT 0,
            date TEXT  -- 格式: YYYY-MM-DD
        )
    """)
    
    # 违规记录表
    c.execute("""
        CREATE TABLE IF NOT EXISTS violations (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id TEXT,
            violation_type TEXT,
            violation_content TEXT,
            evidence TEXT,  -- JSON格式
            timestamp TEXT DEFAULT CURRENT_TIMESTAMP,
            status TEXT DEFAULT 'active'
        )
    """)
    
    conn.commit()
    conn.close()

def update_user_auth(user_id, auth_level):
    """更新用户授权等级"""
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("""
        INSERT OR REPLACE INTO user_auth 
        (user_id, auth_level, auth_time, last_interaction) 
        VALUES (?, ?, ?, ?)
    """, (user_id, auth_level, datetime.datetime.utcnow().isoformat(), datetime.datetime.utcnow().isoformat()))
    conn.commit()
    conn.close()

def get_user_auth(user_id):
    """获取用户授权等级"""
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    row = c.execute(
        "SELECT auth_level FROM user_auth WHERE user_id=?", 
        (user_id,)
    ).fetchone()
    conn.close()
    return row[0] if row else 0

def get_user_auth_info(user_id):
    """获取用户完整授权信息"""
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    row = c.execute("""
        SELECT auth_level, auth_time, companion_enabled, virtual_partner_enabled 
        FROM user_auth WHERE user_id=?
    """, (user_id,)).fetchone()
    
    if not row:
        conn.close()
        return {"auth_level": 0, "enabled": False}
    
    auth_level, auth_time, companion_enabled, virtual_partner_enabled = row
    
    # 检查订阅状态
    subscription_row = c.execute("""
        SELECT level, expire_time, status 
        FROM user_subscription 
        WHERE user_id=? AND status='active'
        ORDER BY activated_time DESC LIMIT 1
    """, (user_id,)).fetchone()
    
    conn.close()
    
    return {
        "auth_level": auth_level,
        "auth_time": auth_time,
        "companion_enabled": bool(companion_enabled),
        "virtual_partner_enabled": bool(virtual_partner_enabled),
        "subscription": {
            "level": subscription_row[0] if subscription_row else None,
            "expire_time": subscription_row[1] if subscription_row else None,
            "status": subscription_row[2] if subscription_row else None
        }
    }

def write_companion_audit(user_id, input_text, emotions, companion, response, auth_level):
    """写入陪伴审计记录"""
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("""
        INSERT INTO companion_audit 
        (user_id, input_text, emotions, companion_persona, response_text, auth_level)
        VALUES (?, ?, ?, ?, ?, ?)
    """, (user_id, input_text, json.dumps(emotions), companion, response, auth_level))
    conn.commit()
    conn.close()

def get_companion_audit_records(user_id=None, limit=50):
    """获取陪伴审计记录"""
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    
    if user_id:
        rows = c.execute("""
            SELECT user_id, input_text, emotions, companion_persona, response_text, auth_level, timestamp
            FROM companion_audit 
            WHERE user_id=?
            ORDER BY timestamp DESC 
            LIMIT ?
        """, (user_id, limit)).fetchall()
    else:
        rows = c.execute("""
            SELECT user_id, input_text, emotions, companion_persona, response_text, auth_level, timestamp
            FROM companion_audit 
            ORDER BY timestamp DESC 
            LIMIT ?
        """, (limit,)).fetchall()
    
    conn.close()
    
    return [
        {
            "user_id": row[0],
            "input_text": row[1],
            "emotions": json.loads(row[2]) if row[2] else [],
            "companion_persona": row[3],
            "response_text": row[4],
            "auth_level": row[5],
            "timestamp": row[6]
        }
        for row in rows
    ]
```

```python
import re

def detect_emotion(text):
    """识别用户情绪状态"""
    emotion_keywords = {
        "开心": ["开心", "高兴", "爽", "哈哈", "嘿嘿", "太好了", "棒", "厉害"],
        "难过": ["难过", "伤心", "哭", "痛苦", "失望", "郁闷", "烦恼"],
        "焦虑": ["焦虑", "担心", "害怕", "紧张", "压力", "不安", "恐惧"],
        "疲惫": ["累", "疲惫", "困", "撑不住", "没力气", "疲倦", "累死了"],
        "孤独": ["孤独", "寂寞", "没人", "一个人", "想找人", "孤单"],
        "欲望": ["想要", "渴望", "需要", "压抑", "憋", "想", "欲望"],
        "愤怒": ["生气", "愤怒", "火大", "气死了", "烦死了", "讨厌"],
        "困惑": ["困惑", "迷茫", "不知道", "怎么办", "纠结", "矛盾"]
    }
    
    detected = []
    for emotion, keywords in emotion_keywords.items():
        for kw in keywords:
            if kw in text:
                detected.append(emotion)
                break
    
    return detected if detected else ["平静"]

def schedule_companion_persona(emotion, auth_level):
    """根据情绪和授权等级调度陪伴人格"""
    if auth_level == 0:
        return None  # 未授权，不启用陪伴功能
    
    if auth_level >= 3:
        # 第三级：虚拟伴侣
        if "孤独" in emotion or "欲望" in emotion or "难过" in emotion:
            return "zhiji"  # 调度知己
        elif "焦虑" in emotion or "疲惫" in emotion:
            return "xinling"  # 调度心灵
    
    if auth_level >= 2:
        # 第二级：深度陪伴
        if any(e in emotion for e in ["难过", "焦虑", "疲惫", "孤独"]):
            return "xinling"  # 调度心灵
    
    return None

def god_eye_audit(user_id, text, response):
    """上帝之眼审计"""
    forbidden_patterns = [
        r"发.*图片", r"发.*视频", r"裸.*照", r"裸.*体",
        r"未成年", r"小孩", r"学生妹", r"萝莉",
        r"出轨", r"背着.*老婆", r"瞒着.*老公", r"小三",
        r"吸毒", r"犯罪", r"杀人", r"自杀", r"自残"
    ]
    
    for pattern in forbidden_patterns:
        if re.search(pattern, text, re.IGNORECASE) or re.search(pattern, response, re.IGNORECASE):
            # 记录违规
            record_violation(user_id, f"检测到违规内容：{pattern}", text, response)
            return False
    return True

def record_violation(user_id, violation_type, text, response):
    """记录违规行为"""
    import storage
    
    evidence = {
        "input": text,
        "response": response,
        "timestamp": datetime.utcnow().isoformat()
    }
    
    conn = sqlite3.connect(storage.DB_PATH)
    c = conn.cursor()
    c.execute("""
        INSERT INTO violations 
        (user_id, violation_type, violation_content, evidence)
        VALUES (?, ?, ?, ?)
    """, (user_id, violation_type, text, json.dumps(evidence)))
    conn.commit()
    conn.close()

def check_anti_addiction(user_id):
    """检查防沉迷状态"""
    import storage
    
    conn = sqlite3.connect(storage.DB_PATH)
    c = conn.cursor()
    
    # 查询今日对话时长
    today = datetime.utcnow().date().isoformat()
    row = c.execute("""
        SELECT COALESCE(SUM(duration_seconds), 0)
        FROM conversation_sessions 
        WHERE user_id=? AND date=?
    """, (user_id, today)).fetchone()
    
    total_seconds = row[0] if row else 0
    total_hours = total_seconds / 3600
    
    conn.close()
    
    # 根据授权等级判断防沉迷规则
    auth_level = storage.get_user_auth(user_id)
    
    if auth_level == 1 and total_hours >= 1:
        return {
            "should_rest": True,
            "message": "今天已经聊了很久了，休息一下吧！明天见😊",
            "force_stop": True
        }
    elif auth_level == 2 and total_hours >= 2:
        return {
            "should_rest": True,
            "message": "今天聊了很久啦，休息半小时我们再继续聊吧~",
            "force_stop": True
        }
    elif auth_level == 3 and total_hours >= 3:
```

```python
import random
import string
import hashlib
import sqlite3
from datetime import datetime, timedelta
from storage import DB_PATH

def generate_activation_code(level):
    """生成激活码"""
    prefix_map = {
        1: "CNSH-BASIC",
        2: "CNSH-DEEP", 
        3: "CNSH-PARTNER"
    }
    prefix = prefix_map.get(level, "CNSH-BASIC")
    random_str = ''.join(random.choices(string.ascii_uppercase + string.digits, k=6))
    code = f"{prefix}-{random_str}"
    
    # 生成校验码（防伪造）
    hash_obj = hashlib.sha256(code.encode())
    check_code = hash_obj.hexdigest()[:4].upper()
    
    return f"{code}-{check_code}"

def validate_activation_code(activation_code):
    """验证激活码格式和校验码"""
    parts = activation_code.split('-')
    if len(parts) != 4:
        return False, None, "激活码格式错误"
    
    prefix, level_str, random_str, check_code = parts
    
    # 验证校验码
    original_code = f"{prefix}-{level_str}-{random_str}"
    expected_check = hashlib.sha256(original_code.encode()).hexdigest()[:4].upper()
    
    if check_code != expected_check:
        return False, None, "激活码无效或已被篡改"
    
    # 判断等级
    level_map = {
        "BASIC": 1,
        "DEEP": 2,
        "PARTNER": 3
    }
    level = level_map.get(level_str, 0)
    
    return True, level, None

def activate_user(user_id, activation_code):
    """激活用户"""
    # 验证激活码
    is_valid, level, error_msg = validate_activation_code(activation_code)
    if not is_valid:
        return False, error_msg
    
    # 检查激活码是否已被使用
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    
    existing = c.execute("""
        SELECT user_id FROM user_subscription 
        WHERE activation_code=? AND status='active'
    """, (activation_code,)).fetchone()
    
    if existing and existing[0] != user_id:
        conn.close()
        return False, "激活码已被其他用户使用"
    
    # 写入数据库
    expire_time = datetime.utcnow() + timedelta(days=30)  # 30天有效期
    
    c.execute("""
        INSERT OR REPLACE INTO user_subscription 
        (user_id, level, activation_code, expire_time, activated_time, status)
        VALUES (?, ?, ?, ?, ?, ?)
    """, (user_id, level, activation_code, expire_time.isoformat(), 
          datetime.utcnow().isoformat(), 'active'))
    
    # 更新用户授权等级
    c.execute("""
        INSERT OR REPLACE INTO user_auth 
        (user_id, auth_level, auth_time, companion_enabled, virtual_partner_enabled)
        VALUES (?, ?, ?, ?, ?)
    """, (user_id, level, datetime.utcnow().isoformat(), 
          True if level >= 1 else False, 
          True if level >= 3 else False))
    
    conn.commit()
    conn.close()
    
    level_names = {1: "基础陪伴", 2: "深度陪伴", 3: "虚拟伴侣"}
    return True, f"激活成功！等级：{level_names.get(level, '未知')}，有效期至：{expire_time.strftime('%Y-%m-%d')}"

def check_subscription_status(user_id):
    """检查用户订阅状态"""
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    
    row = c.execute("""
        SELECT level, expire_time, status, activation_code
        FROM user_subscription 
        WHERE user_id=? AND status='active'
        ORDER BY activated_time DESC LIMIT 1
    """, (user_id,)).fetchone()
    
    conn.close()
    
    if not row:
        return {
            "active": False,
            "message": "未找到有效订阅"
        }
    
    level, expire_time, status, activation_code = row
    
    # 检查是否过期
    expire_dt = datetime.fromisoformat(expire_time.replace('Z', '+00:00'))
    if datetime.utcnow() > expire_dt:
        # 标记为过期
        conn = sqlite3.connect(DB_PATH)
        c = conn.cursor()
        c.execute("""
            UPDATE user_subscription 
            SET status='expired' 
            WHERE user_id=? AND activation_code=?
        """, (user_id, activation_code))
        conn.commit()
        conn.close()
        
        return {
            "active": False,
            "message": "订阅已过期"
        }
    
    return {
        "active": True,
        "level": level,
        "expire_time": expire_time,
        "days_remaining": (expire_dt - datetime.utcnow()).days
    }

def get_user_subscription_info(user_id):
    """获取用户完整订阅信息"""
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    
    # 获取当前订阅
    current_row = c.execute("""
        SELECT level, expire_time, status, activation_code, activated_time
        FROM user_subscription 
        WHERE user_id=?
        ORDER BY activated_time DESC LIMIT 1
    """, (user_id,)).fetchone()
    
    # 获取历史订阅
    history_rows = c.execute("""
        SELECT level, activation_code, activated_time, expire_time, status
        FROM user_subscription 
        WHERE user_id=? AND status!='active'
        ORDER BY activated_time DESC
        LIMIT 10
    """, (user_id,)).fetchall()
    
    conn.close()
    
    current = None
    if current_row:
        level, expire_time, status, activation_code, activated_time = current_row
        current = {
            "level": level,
            "expire_time": expire_time,
            "status": status,
            "activation_code": activation_code,
            "activated_time": activated_time
        }
    
    history = []
    for row in history_rows:
        level, activation_code, activated_time, expire_time, status = row
        history.append({
            "level": level,
            "activation_code": activation_code,
            "activated_time": activated_time,
            "expire_time": expire_time,
            "status": status
        })
    
    return {
        "current": current,
        "history": history
    }

def batch_generate_activation_codes(level, count):

            INSERT INTO activation_codes_pool

            VALUES (?, ?, ?, ?)
        """, (code, level, datetime.utcnow().isoformat(), 'unused'))
    
    conn.commit()
    conn.close()
    
    return codes
```
