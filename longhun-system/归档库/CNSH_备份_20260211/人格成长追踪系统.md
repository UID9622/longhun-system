# 人格成长优化算法
def optimize_persona_growth(persona_id, current_skills):
    """根据当前能力优化成长路径"""
    
    # 分析短板
    weakest_skill = min(current_skills, key=current_skills.get)

    recommendations = {
        "知识库": "投喂专业知识卡片",

        "创新思维": "进行创造性挑战",

    return {
        "短板分析": weakest_skill,
        "成长建议": recommendations[weakest_skill],
        "预期提升": "+2分/周"
    }
