```
龍魂决策 = max((全球收益 × W_卦象 × W_文化) / (群体损失 + ε_护弱))

class 侦察兵权重决策器:
    """
    侦察兵的核心职责：信息收集、数据调研
    对应卦象：艮卦（止静观察）
    """
    
    def __init__(self):
        self.本命卦象 = "艮卦"  # ☶ 止静观察
        self.基础权重 = {"个体": 0.3, "群体": 0.3, "全球": 0.4}
        self.护弱关键词 = ["隐私泄露", "个人数据", "用户隐私"]
    
    def 信息收集决策(self, 信息源列表):
        """
        应用龍魂权重算法进行信息收集决策
        核心原则：不侵犯隐私，不收集敏感个人信息
        """
        决策报告 = []
        
        for 信息源 in 信息源列表:
            # 第一步：隐私合规检查（甲骨文护弱者）
            护弱系数, 审计颜色 = self.检查隐私合规(信息源)
            
            # 如果触发红色熔断，直接跳过
            if 审计颜色 == "🔴":
                决策报告.append({
                    "信息源": 信息源["名称"],
                    "决策": "❌ 阻断：涉及隐私侵犯",
                    "审计结果": "🔴 红色熔断",
                    "理由": "甲骨文护弱者修正：禁止收集个人隐私数据"
                })
                continue
            
            # 第二步：信息价值评估
            信息价值 = self.评估信息价值(信息源)
            收集成本 = self.计算收集成本(信息源)
            风险系数 = self.评估风险系数(信息源)
            
            # 第三步：获取当前卦象
            当前卦象 = self.获取时辰卦象()
            卦象权重 = self.获取卦象权重(当前卦象)
            
            # 第四步：龍魂决策计算
            加权全球收益 = 信息价值["系统安全价值"] * 卦象权重["全球"]
            加权群体收益 = 信息价值["知识扩展价值"] * 卦象权重["群体"]
            修正后损失 = 收集成本 * 风险系数 + (1.0 / 护弱系数)
            
            if 修正后损失 > 0:
                收益损失比 = (加权全球收益 + 加权群体收益) / 修正后损失
            else:
                收益损失比 = float('inf')
            
            # 第五步：三色审计决策
            if 收益损失比 > 3.0 and 审计颜色 == "🟢":
                决策 = "✅ 建议收集"
                审计结果 = "🟢 绿色通过"
            elif 收益损失比 > 1.5 or 审计颜色 == "🟡":
                决策 = "🟡 需要确认"
                审计结果 = "🟡 黄色确认"
            else:
                决策 = "❌ 不建议收集"
                审计结果 = "🔴 红色熔断（低风险收益比）"
            
            决策报告.append({
                "信息源": 信息源["名称"],
                "卦象": 当前卦象,
                "信息价值": 信息价值,
                "收益损失比": round(收益损失比, 2),
                "决策": 决策,
                "审计结果": 审计结果
            })
        
        return 决策报告
```
