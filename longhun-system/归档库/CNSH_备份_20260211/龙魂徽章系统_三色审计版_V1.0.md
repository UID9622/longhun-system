### **问题2：徽章生成前未强制三色审计**

```yaml
问题描述:
  - 徽章不能随便给
  - 必须先三色审计
  - 红色内容不能发徽章

解决方案:
  生成徽章前强制三色审计

审计规则:
  🔴 红色 → 拒绝生成徽章
     原因: 项目含违法内容，不符合国产认证标准
  
  🟡 黄色 → 生成但标记"⚠️需人工审核"
     原因: 有警告内容，需人工复审
  
  🟢 绿色 → 直接生成徽章
     原因: 内容安全，符合标准
```

**实现代码**：

```javascript
// 徽章生成前强制三色审计
function 生成徽章(项目, 徽章类型) {
    
    // 第1步：三色审计
    const 审计结果 = 三色审计.检查(项目);
    
    // 第2步：根据审计结果决定是否生成
    if (审计结果 === "红色") {
        return {
            成功: false,
            原因: "项目含违法内容，不符合国产认证标准",
            徽章: null
        };
    }
    
    if (审计结果 === "黄色") {
        // 生成徽章，但添加警告标签
        const 徽章 = 生成基础徽章(项目, 徽章类型);
        徽章.添加标签("⚠️需人工审核");
        return {
            成功: true,
            原因: "黄色警告，需人工审核",
            徽章: 徽章
        };
    }
    
    if (审计结果 === "绿色") {
        // 直接生成徽章
        const 徽章 = 生成基础徽章(项目, 徽章类型);
        return {
            成功: true,
            原因: "绿色通过",
            徽章: 徽章
        };
    }
}

// 示例：生成"国产认证"徽章
const 结果 = 生成徽章(某项目, "国产认证");
if (结果.成功) {
    显示徽章(结果.徽章);
} else {
    拒绝并告知原因(结果.原因);
}
```
