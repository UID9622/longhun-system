### 2. 优化设备检测逻辑

**当前问题**:
- 设备检测失败时会直接报错
- 缺少详细的错误信息
- 没有备用检测方案

**改进方案**:

```python
def _test_device_detection(self) -> Dict[str, Any]:
    """测试设备检测 - 改进版"""
    results = {}

    # 摄像头检测（多方案）
    try:
        import cv2
        cap = cv2.VideoCapture(0)
        if cap.isOpened():
            # 尝试获取一帧以确认摄像头工作
            ret, frame = cap.read()
            if ret:
                width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
                height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
                results["camera"] = f"Available ({width}x{height})"
                cap.release()
            else:
                results["camera"] = "Available but no frame"
        else:
            results["camera"] = "Not Available"
    except ImportError:
        results["camera"] = "OpenCV not installed (建议: pip install opencv-python)"
    except Exception as e:
        results["camera"] = f"Error: {str(e)} (建议: 检查摄像头权限)"

    # 音频设备检测（多方案）
    try:
        import pyaudio
        pa = pyaudio.PyAudio()
        device_count = pa.get_device_count()

        # 检查输入设备
        input_devices = []
        output_devices = []
        for i in range(device_count):
            info = pa.get_device_info_by_index(i)
            if info['maxInputChannels'] > 0:
                input_devices.append(info['name'])
            if info['maxOutputChannels'] > 0:
                output_devices.append(info['name'])

        pa.terminate()
        results["audio"] = {
            "input_devices": len(input_devices),
            "output_devices": len(output_devices),
            "total": device_count
        }
    except ImportError:
        results["audio"] = "PyAudio not installed (建议: pip install pyaudio)"
    except Exception as e:
        results["audio"] = f"Error: {str(e)} (建议: 检查音频权限)"

    # 判断状态
    camera_ok = "Available" in str(results.get("camera", ""))
    audio_ok = isinstance(results.get("audio"), dict) and results["audio"]["total"] > 0

    if camera_ok and audio_ok:
        status = "PASS"
    elif camera_ok or audio_ok:
        status = "PARTIAL"
    else:
        status = "FAIL"

    return {
        "status": status,
        "details": results
    }
```

```python
def _check_graphics_compatibility(self) -> Dict[str, Any]:
    """检测图形框架兼容性 - 跨平台版本"""
    results = {}

    if sys.platform == "win32":
        # Windows: DirectX检测
        results["platform"] = "Windows"
        results["framework"] = "DirectX"
        results["status"] = self._check_directx()

    elif sys.platform == "darwin":
        # macOS: Metal框架检测
        results["platform"] = "macOS"
        results["framework"] = "Metal"

        try:
            # 检查Metal支持
            import subprocess
            cmd = "system_profiler SPDisplaysDataType | grep 'Metal'"
            output = subprocess.check_output(cmd, shell=True, text=True)
            metal_supported = "Supported" in output or "Yes" in output

            if metal_supported:
                results["metal_support"] = "Yes"
                results["status"] = "PASS"
            else:
                results["metal_support"] = "No"
                results["status"] = "FAIL"
        except Exception as e:
            results["status"] = "ERROR"
            results["message"] = str(e)

    elif sys.platform.startswith("linux"):
        # Linux: OpenGL检测
        results["platform"] = "Linux"
        results["framework"] = "OpenGL"

        try:
            # 检查OpenGL版本
            import subprocess
            cmd = "glxinfo 2>/dev/null | grep 'OpenGL version'"
            output = subprocess.check_output(cmd, shell=True, text=True)

            if output.strip():
                results["opengl_version"] = output.strip()
                results["status"] = "PASS"
            else:
                results["status"] = "FAIL"
                results["message"] = "glxinfo not available"
        except Exception as e:
            results["status"] = "ERROR"
            results["message"] = str(e)

    else:
        results["platform"] = "Unknown"
        results["status"] = "SKIP"
        results["message"] = f"不支持的平台: {sys.platform}"

    return results
```

### 4. 添加跨平台依赖检测

**改进方案**:

```python
def _check_native_dependencies(self) -> Dict[str, Any]:
    """检查本地系统依赖 - 跨平台"""
    results = {}

    if sys.platform == "win32":
        # Windows: DLL检测
        required_dlls = ["dxgi.dll", "d3d11.dll", "d3d9.dll"]
        system_paths = [
            os.path.join(os.environ.get("SystemRoot", "C:\\Windows"), "System32"),
            os.path.join(os.environ.get("SystemRoot", "C:\\Windows"), "SysWOW64")
        ]

        for dll in required_dlls:
            found = False
            for path in system_paths:
                if os.path.exists(os.path.join(path, dll)):
                    found = True
                    break
            results[dll] = "Found" if found else "Not Found"

    elif sys.platform == "darwin":
        # macOS: Framework检测
        frameworks = [
            "/System/Library/Frameworks/CoreMedia.framework",
            "/System/Library/Frameworks/AVFoundation.framework",
            "/System/Library/Frameworks/Metal.framework"
        ]

        for fw in frameworks:
            fw_name = os.path.basename(fw)
            results[fw_name] = "Found" if os.path.exists(fw) else "Not Found"

    elif sys.platform.startswith("linux"):
        # Linux: 库文件检测
        libraries = ["libGL.so", "libEGL.so", "libv4l2.so"]

        for lib in libraries:
            try:
                cmd = f"ldconfig -p | grep {lib}"
                output = subprocess.check_output(cmd, shell=True, text=True)
                results[lib] = "Found" if output.strip() else "Not Found"
            except:
                results[lib] = "Not Found"

    found_count = sum(1 for v in results.values() if v == "Found")
    total_count = len(results)

    return {
        "status": "PASS" if found_count >= total_count * 0.7 else "FAIL",
        "details": {
            "platform": sys.platform,
            "dependencies": results,
            "summary": f"{found_count}/{total_count} dependencies found"
        }
    }
```

```python
class RTMPStreamTester:
    """RTMP推流测试器"""

    def __init__(self):
        self.test_stream_url = "rtmp://127.0.0.1/live/test"

    def test_stream_push(self) -> Dict[str, Any]:
        """测试推流功能"""
        try:
            import cv2

            # 使用本地摄像头
            cap = cv2.VideoCapture(0)

            # 配置FFmpeg推流
            import subprocess
            cmd = [
                "ffmpeg",
                "-i", "-",
                "-c:v", "libx264",
                "-preset", "ultrafast",
                "-f", "flv",
                self.test_stream_url
            ]

            process = subprocess.Popen(cmd, stdin=subprocess.PIPE)

            # 推送10帧测试
            for _ in range(10):
                ret, frame = cap.read()
                if ret:
                    process.stdin.write(cv2.imencode('.jpg', frame)[1].tobytes())

            process.terminate()
            cap.release()

            return {
                "status": "PASS",
                "message": "推流测试成功"
            }

        except Exception as e:
            return {
                "status": "FAIL",
                "message": f"推流测试失败: {str(e)}"
            }
```

### 7. 开发自动依赖安装功能

**目标**: 一键安装所有测试依赖

```python
class DependencyInstaller:
    """依赖自动安装器"""

    @staticmethod
    def install_all() -> Dict[str, Any]:
        """安装所有必需依赖"""
        results = {
            "success": True,
            "installed": [],
            "failed": []
        }

        dependencies = {
            "opencv-python-headless": "摄像头检测",
            "pyaudio": "音频设备检测",
            "requests": "网络测试"
        }

        for package, purpose in dependencies.items():
            try:
                subprocess.check_call(
                    [sys.executable, "-m", "pip", "install", package],
                    stdout=subprocess.DEVNULL
                )
                results["installed"].append({
                    "package": package,
                    "purpose": purpose
                })
            except:
                results["failed"].append({
                    "package": package,
                    "purpose": purpose
                })
                results["success"] = False

        return results
```
