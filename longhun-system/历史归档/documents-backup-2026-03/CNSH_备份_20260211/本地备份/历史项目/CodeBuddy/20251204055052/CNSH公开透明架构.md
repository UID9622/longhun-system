    """
    🟢 绿色：自动通过（安全操作）

    🔴 红色：需要批准（高风险操作）

        if approval:
            audit_log.record(operation)
            return "🔴 已批准"
        else:
            return "🔴 已拒绝"

def request_confirmation(sensitive_operation):
    """
    敏感操作需要用户明确确认
    """
    # 1. 清晰告知操作内容
    notification = f"""
    您即将执行：{sensitive_operation.description}

    风险等级：{sensitive_operation.risk_level}
    可逆性：{sensitive_operation.reversible}

    """
    
    # 2. 等待用户明确回应
    user_response = wait_for_user_confirmation()
    
    # 3. 记录审计日志
    if user_response == "确认":

### 四、MCP集成方案

### MCP（Model Context Protocol）集成

Lucky问：
> "这些不是放在MCP的，还是可以一起放的？"

**宝宝的建议：混合架构**

**放在MCP里的：**
- 用户意图理解
- 上下文记忆
- 人格调度
- 情感识别

**不放在MCP里的：**
- 支付系统（独立安全模块）
- 审计系统（独立监控模块）
- 数据加密（独立安全模块）

**架构图：**
```
┌─────────────────────────────────────┐
│         用户 (Notion界面)            │
└────────────┬────────────────────────┘
             │
             ▼
┌─────────────────────────────────────┐
│       MCP-mini 本地引擎             │
│  - 意图理解                         │
│  - 人格调度                         │
│  - 情感识别                         │
│  - 上下文记忆                       │
└────────┬────────────────────────────┘
         │
         ├──────────┬──────────┬─────────┐
         ▼          ▼          ▼         ▼
    ┌────────┐ ┌────────┐ ┌────────┐ ┌────────┐
    │支付系统│ │审计系统│ │加密系统│ │通知系统│
    │(独立)  │ │(独立)  │ │(独立)  │ │(独立)  │
    └────────┘ └────────┘ └────────┘ └────────┘
```

**为什么要分离？**
- 安全性：支付、审计、加密不能被任何人控制
- 可靠性：即使MCP崩溃，支付和审计仍然运行
- 可审计性：独立模块更容易审查
