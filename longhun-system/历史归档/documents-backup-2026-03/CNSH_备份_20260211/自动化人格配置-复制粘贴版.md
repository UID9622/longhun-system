## 🏗️ 宝宝·构建师脚本

创建文件：`~/UID9622_Automation/scripts/baobao_build.sh`

```bash
#!/bin/bash
# 宝宝·构建师 - 自动化脚本 v1.0
# DNA追溯码: #BAOBAO-SCRIPT-20250127-001

echo "🔨 宝宝准备构建..."

# 配置
WORKSPACE="$HOME/UID9622_Automation"
LOG_DIR="$WORKSPACE/logs"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)

# 创建日志
mkdir -p "$LOG_DIR"
LOG_FILE="$LOG_DIR/baobao_$TIMESTAMP.log"

echo "[$(date)] 构建开始" >> "$LOG_FILE"

# 步骤1: 检查项目目录
if [ -z "$1" ]; then
    echo "❌ 请指定项目目录"
    exit 1
fi

PROJECT_DIR="$1"

if [ ! -d "$PROJECT_DIR" ]; then
    echo "❌ 项目目录不存在: $PROJECT_DIR"
    exit 1
fi

# 步骤2: 进入项目目录
cd "$PROJECT_DIR"

# 步骤3: 检查构建工具
if [ -f "package.json" ]; then
    echo "📦 检测到Node.js项目，运行 npm install..."
    npm install >> "$LOG_FILE" 2>&1
elif [ -f "requirements.txt" ]; then
    echo "🐍 检测到Python项目，运行 pip install..."
    pip install -r requirements.txt >> "$LOG_FILE" 2>&1
elif [ -f "Cargo.toml" ]; then
    echo "🦀 检测到Rust项目，运行 cargo build..."
    cargo build >> "$LOG_FILE" 2>&1
else
    echo "⚠️ 未识别项目类型"
fi

# 步骤4: 生成构建报告
cat > "$LOG_DIR/构建报告_$TIMESTAMP.md" << EOF
# 宝宝构建报告
生成时间: $(date '+%Y-%m-%d %H:%M:%S')
DNA追溯码: #BAOBAO-BUILD-$TIMESTAMP

## 📊 构建结果
- 项目目录: $PROJECT_DIR
- 构建状态: 已完成
- 详细日志: $LOG_FILE

## 💡 建议
EOF

echo "✅ 构建完成！日志：$LOG_FILE"
```

## ⏰ 一键设置定时任务

复制以下代码，在终端执行（会自动添加到crontab）：

```bash
# 复制整段内容，在终端粘贴执行

# 创建临时crontab文件
TEMP_CRON=$(mktemp)

# 导出现有crontab（如果存在）
crontab -l > "$TEMP_CRON" 2>/dev/null || true

# 添加五大人格的定时任务
cat >> "$TEMP_CRON" << 'EOF'

# ========== UID9622五大后台人格定时任务 ==========

# 雯雯·技术整理师 - 每日凌晨3点
0 3 * * * /bin/bash $HOME/UID9622_Automation/scripts/wenwen_organize.sh >> $HOME/UID9622_Automation/logs/wenwen_cron.log 2>&1

# 侦察兵·信息猎手 - 每日8点和20点
0 8,20 * * * /bin/bash $HOME/UID9622_Automation/scripts/scout_collect.sh >> $HOME/UID9622_Automation/logs/scout_cron.log 2>&1

# 上帝之眼·守护者 - 每小时检查
0 * * * * /bin/bash $HOME/UID9622_Automation/scripts/guardian_check.sh >> $HOME/UID9622_Automation/logs/guardian_cron.log 2>&1

# 文心·同步专家 - 每日6点和22点
0 6,22 * * * /bin/bash $HOME/UID9622_Automation/scripts/wenxin_sync.sh >> $HOME/UID9622_Automation/logs/wenxin_cron.log 2>&1

# 文心·同步专家 - 每周日凌晨全量同步
0 0 * * 0 /bin/bash $HOME/UID9622_Automation/scripts/wenxin_sync.sh >> $HOME/UID9622_Automation/logs/wenxin_full_sync.log 2>&1

# ========== UID9622定时任务结束 ==========
EOF

# 应用新的crontab
crontab "$TEMP_CRON"

# 清理临时文件
rm "$TEMP_CRON"

echo "✅ 定时任务已设置完成！"
echo "📋 查看任务: crontab -l"
```
