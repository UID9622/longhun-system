### 1.2 识别算法

```python
def identify_scene(user_input: str) -> dict:
    """
    场景识别算法
    
    返回: {
        "scene": "A-G",
        "emotion": "🟢🟡🔴",
        "urgency": 0-10,
        "matched_keywords": [...],
        "confidence": 0-1
    }
    """
    
    # 关键词匹配
    keywords_map = {
        "A": ["选", "纠结", "还是", "哪个好"],
        "B": ["完了", "崩", "撑不住", "放弃"],
        "C": ["稳了", "必赢", "起飞", "顺"],
        "D": ["卡", "做不", "没进展", "推不动"],
        "E": ["为什么", "怎么理解", "不懂", "道理"],
        "F": ["他", "关系", "不理解", "矛盾"],
        "G": ["布局", "长远", "规划", "下一步"]
    }
    
    # 情绪词识别
    emotion_keywords = {
        "🔴": ["完了", "崩", "绝望", "没意义"],
        "🟡": ["纠结", "迷茫", "着急", "困惑"],
        "🟢": ["好奇", "想知道", "规划", "布局"]
    }
    
    # 紧急度判断（0-10）
    urgency_keywords = {
        10: ["急", "马上", "立即", "现在"],
        7: ["尽快", "赶紧", "快"],
        3: ["有空", "方便", "慢慢"]
    }
    
    # 执行匹配...
    scene = match_scene(user_input, keywords_map)
    emotion = match_emotion(user_input, emotion_keywords)
    urgency = calculate_urgency(user_input, urgency_keywords)
    
    return {
        "scene": scene,
        "emotion": emotion,
        "urgency": urgency,
        "confidence": calculate_confidence(scene, emotion)
    }
```
